掘金更改动态头像

打开修改个人信息页面 , f12 , 打开console, 输入下面的代码 ,回车 , 刷新看一下吧

注释：上传个图片看看自己的aid是不是2608，如果不是改成你的aid

var ajax = new XMLHttpRequest();
      ajax.open("post", "https://juejin.cn/web/user/update/user_info/", true);
      ajax.setRequestHeader(
        "content-type",
        "application/x-www-form-urlencoded"
      );
      ajax.send(
        "aid=2608&avatar=https://p3-passport.byteimg.com/img/user-avatar/2741f39cc0043a66c7e568a982d2647a~100x100.awebp"
      );
      ajax.onreadystatechange = function() {
        if (ajax.readyState == 4 && ajax.status == 200) {
          var c = ajax.responseText;
          console.log(c);
        }
      };

function jjPost(){ var ajax = new XMLHttpRequest();
      ajax.open("post", "https://juejin.cn/web/user/update/user_info/", true);
      ajax.setRequestHeader(
        "content-type",
        "application/x-www-form-urlencoded"
      );
      ajax.send(
        "aid=2608&avatar=https://p3-passport.byteimg.com/img/user-avatar/2741f39cc0043a66c7e568a982d2647a~100x100.awebp"
      );
      ajax.onreadystatechange = function() {
        if (ajax.readyState == 4 && ajax.status == 200) {
          var c = ajax.responseText;
          console.log(c);
        }
      };}
setInterval(jjPost,1000)
