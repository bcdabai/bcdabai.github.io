c语言图形打印 12345

#include<stdio.h>
int main()
{
    int i,d,n;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        for(d=n;d>i;d--)
        {
            printf(" ");
        }
        for(d=1;d<=i;d++)
        {
            printf("%d",d);
        }
        for(d=i-1;d>=1;d--)
        {
            printf("%d",d);
        }
        printf("\n");
    }
    for(i=1;i<=n-1;i++)
    {
        for(d=1;d<=i;d++)
        {
            printf(" ");
        }
        for(d=1;d<=n-i;d++)    \\  注意1 和 n-i
        {
            printf("%d",d);
        }
        for(d=n-i-1;d>=1;d--)
        {
            printf("%d",d);
        }
        printf("\n");
    }
    return 0;
}
