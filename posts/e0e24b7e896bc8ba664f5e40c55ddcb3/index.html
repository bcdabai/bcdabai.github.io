#include<stdio.h>
int main(void)
{
    printf("Here's one way to print a ");
    printf("long string.\n");

long string.\n");
    printf("Here's the newest way to print a " "long string.\n");
    printf("谢谢你对我的帮助！");
    return 0;
}
