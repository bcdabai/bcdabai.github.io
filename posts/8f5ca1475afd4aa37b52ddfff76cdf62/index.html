python 自定义时间 加减天数

import datetime



# 现在的时间
setTime ='2015-04-16'
now=datetime.datetime.strptime(setTime,'%Y-%m-%d')
print(now)
# 递增的时间
delta = datetime.timedelta(days=1)
# 88天后的时间
endnow = now + datetime.timedelta(days=88)#设置小时的就改成hours
# 六天后的时间转换成字符串
endnow = str(endnow.strftime('%Y-%m-%d'))
offset = now
# 当日期增加到88天后的日期，循环结束
while str(offset.strftime('%Y-%m-%d')) != endnow:
    offset += delta
    print(str(offset.strftime('%Y-%m-%d')))
