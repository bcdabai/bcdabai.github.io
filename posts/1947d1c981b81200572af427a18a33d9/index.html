bool KafkaResourceManager::Register(std::string name, std::function<std::shared_ptr<rubik::KafkaResource>()> callback) {
  std::shared_ptr<rubik::KafkaResource> resource = callback();
  if (resource) {
    return false;
  }
  kafka_resources_[name] = resource;
  return true;
}
