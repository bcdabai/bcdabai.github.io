计算1！+2！+. . . +10！的值

#include <stdio.h>
#include <stdlib.h>
int main()
{
int n = 0;
int ret = 1;
int sum = 0;
for (n = 1; n <= 10; n++)
{
ret = ret*n;

	sum = sum + ret;
}
printf("sum=%d\n ", sum);

system(“pause”);
return 0;
}
