GO创建数组

func main() {
	var numbers1[5] int = [5]int {1,2,3,4,5}
	fmt.Println(numbers1[2])

	numbers2 := [5]int{1,2}
	fmt.Println(numbers2[4])

	numbers3 := [5]int{2:4,4:10}
	fmt.Println(numbers3[4])

	numbers4 := [...]int{7,8,9}
	fmt.Println(len(numbers4))
	fmt.Println(numbers4[2])

	var numbers5[7] int
	numbers5[3] = 100
	numbers5[4] = 200
	fmt.Println(len(numbers5))
	fmt.Println(numbers5[2])
}
