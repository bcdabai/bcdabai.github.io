怎么用C语言在屏幕上打印数字

#include <stdio.h>
#include <string.h>

int main(void)
{
    char arr1[] = { "jin tian ye yao jia you a cheng jia jia" };
    char arr2[] = { "#########################################" };
    int left = 0;
    int right = strlen(arr1) - 1;
    while (left <= right)
    {
        arr2[left] = arr1[left];
        arr2[right] = arr1[right];
        printf("%s\n",arr2);
        Sleep(1000);   //睡眠一秒；
        system("cls");  //清空屏幕；
        left++;
        right--;
    }
    printf("%s\n", arr2);
    return 0;
}
