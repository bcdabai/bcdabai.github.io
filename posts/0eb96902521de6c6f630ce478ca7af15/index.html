杨辉三角的原理

public class Demo08 {
    //杨辉三角
    public static void main(String[] args) {
        //打印三角形 5行
        for (int i = 1; i <= 5; i++) {
            //第一部分，左边空部分
            for (int j = 5; j >= i; j--) {
                System.out.print(" ");
            }
//            第二部分，左三角
            for (int j = 1; j <= i; j++) {
                System.out.print("*");
            }
            //第三部分，右三角
            for (int j = 1; j < i; j++) {
                System.out.print("*");
            }
            System.out.println();
        }
    }
}
