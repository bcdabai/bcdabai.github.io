	public static int treeDepth(TreeNode root) {

			return 0;
		}
		return Math.max(treeDepth(root.left), treeDepth(root.right))+1;
	}

		if (root==null) {
			return 0;
		}

		else return treeLeaf(root.left)+treeLeaf(root.right);
	}
