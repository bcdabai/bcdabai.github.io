    ne[i] = j;

例题

求出模板串P在模式串S中所有出现的位置的起始下标。

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int N = 10010, M = 1010;

int n, m;
char s[N], p[M];
int ne[M];

int main()
{
    //s模式串， p匹配串
    cin >> n >> s + 1 >> m >> p + 1;

    for (int i = 2, j = 0; i <= m; i++)
    {
        while (j && p[i] != p[j + 1])
            j = ne[j];
        if (p[i] == p[j + 1])
            j++;
        ne[i] = j;
    }

    for (int i = 1, j = 0; i <= n; i++)
    {
        while (j && s[i] != p[j + 1])
            j = ne[j];
        if (s[i] == p[j + 1])
            j++;
        if (j == m)
        {
            cout << i - m + 1 << ' ';
            j = ne[j];
        }
    }

    return 0;
}
