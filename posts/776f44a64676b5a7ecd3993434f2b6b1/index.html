打印金字塔，1 121 12321 1234321 123454321

小程序示例，代码如下：

void main()
{
    int i,j,k,m;
    for(i=1;i<=5;i++){          //打印行数,共5行
        for(j=1;j<=5-i;j++){
            printf(" ");        //打印每行的空格数
        }
        for(k=1;k<=i;k++){      //打印前半部分
            printf("%d",k);
        }
        for(m=i-1;m>0;m--){     //打印后半部分
            printf("%d",m);
        }
        printf("\n");           //换行
    }
    system("pause");
}
