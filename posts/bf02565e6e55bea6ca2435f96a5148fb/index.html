        cout << "文件读取失败！" << endl;

    }
    
    //跳过文件中第一行表头

    while (!FilePtr.eof()) {
        FilePtr >> L.elem[v].id >> L.elem[v].company_name 
        >>L.elem[v].product_name >> L.elem[v].product_price
        >> L.elem[v].order_num >> L.elem[v].order_price;
