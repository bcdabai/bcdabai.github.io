js:利用for循环，输出1-10所有数的平方和(即1-10的阶乘)

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>输出1-10所有数的平方和</title>
</head>
<body>

</body>
<script>
    var i;
    var j;
    var sum = 0;
    for(i = 1;i <= 10;i++){
          j= i*i;
        sum = sum + j;
    }
    alert(sum)
</script>
</html>
