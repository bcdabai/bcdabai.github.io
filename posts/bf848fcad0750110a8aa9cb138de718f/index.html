编写一个递归函数，将输入的整数按照它的逆序输出，如输入12345，输出54321

#include<iostream>
 using namespace std;
 void print(int a)
 {
 	if(!a)
 	{
 		return;
	}
	cout<<a%10;
	print(a/10);
 	 
 }
int main()
 {
 	int n;
 	cin>>n;
 	print(n);
 	return 0;
 }
