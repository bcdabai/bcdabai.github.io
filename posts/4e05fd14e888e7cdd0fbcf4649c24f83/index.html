配置sass全局变量Undefined variable: “$theme-color“.

原因：sass版本不同选项不同

sass v8 版本使用 prependData

module.exports = {
    css: {
        loaderOptions: {
           sass: {
                    prependData: `@import "assets/css/_variable.scss";`   
                }
           }
    },
    configureWebpack: {
        resolve: {
            alias: {
                'common': '@/common',
                'components': '@/components',
                'views': '@/views',
                'network': '@/network',
                'assets': '@/assets'
            }
        }
    }
}

之前的版本使用：

module.exports = {
    css: {
        loaderOptions: {
           sass: {
                    data: `@import "assets/css/_variable.scss";`   
                    // sass v8版本使用这个  prependData
                }
           }
    },
    configureWebpack: {
        resolve: {
            alias: {
                'common': '@/common',
                'components': '@/components',
                'views': '@/views',
                'network': '@/network',
                'assets': '@/assets'
            }
        }
    }
}
