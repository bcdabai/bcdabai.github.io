Ubuntu20 USB 权限配置

//

（保姆级）Ubuntu下的USB设备永久权限+设备起别名方法

cd /etc/udev/rules.d

sudo vim motor_control_usb.rules

增加如下内容


SUBSYSTEMS=="usb", ATTRS{idVendor}=="0483", ATTRS{idProduct}=="2000", GROUP="edda", MODE="0777"
SUBSYSTEMS=="usb", ATTRS{idVendor}=="0483", ATTRS{idProduct}=="2001", GROUP="edda", MODE="0777"
SUBSYSTEMS=="usb", ATTRS{idVendor}=="0483", ATTRS{idProduct}=="2002", GROUP="edda", MODE="0777"
SUBSYSTEMS=="usb", ATTRS{idVendor}=="0483", ATTRS{idProduct}=="2003", GROUP="edda", MODE="0777"
SUBSYSTEMS=="usb", ATTRS{idVendor}=="0483", ATTRS{idProduct}=="2004", GROUP="edda", MODE="0777"
SUBSYSTEMS=="usb", ATTRS{idVendor}=="0483", ATTRS{idProduct}=="2005", GROUP="edda", MODE="0777"                                                                                           


立即生效

sudo udevadm trigger
