返回顶部代码

　　<dic id='scrollTop'>返回顶部</div>

<script>

　　$(function(){

　　  $(function(){

　       $(window).scroll(function(){

　　　　if($(window.sctollTop()>150){

　　　　  $("#scrollTop").fadeIn(1000);

　　}else{

　　　　  $("#scrollTop").fadeOut(1000);

    }

    });　

  });

　 //返回顶部动作

　　$("#scrollTop").click(function(){

　　　　$(body,html).animate({scrollTop:0},1000);

　　});

});

</script>
