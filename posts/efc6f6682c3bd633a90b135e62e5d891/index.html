Script部分:

    var but = document.querySelectorAll(".theme_box")
    var bigbox = document.querySelector(".bigBox")
    var shadow = document.querySelectorAll(".shadow_color")
    var themeOptions = {
        '白色主题': {
            'boxClass': 'bigBox2',
            'shadowClass': 'shadow_color2'
        },
        '黑色主题': {
            'boxClass': 'bigBox',
            'shadowClass': 'shadow_color'
        }
    };

    but.forEach(item => {
        item.addEventListener("click", function (e) {
            var color = e.target.textContent;
            var selectedTheme = themeOptions[color];

            if (selectedTheme) {
                bigbox.className = selectedTheme['boxClass'];
                shadow.forEach(items => {
                    items.classList.remove('shadow_color', 'shadow_color2');
                    items.classList.add(selectedTheme['shadowClass']);
                });
            }
        });
    });
