Java程序①求1!+2!+…+10!的值

初学Java，小小小小程序，练个手。☀☀☀

/**
 * 编写应用程序求1！+2！+……+10！
 * @author 
 *
 */
public class Sum {

	public static void main(String[] args) {
		int i = 1;
		int a = 1;
		int sum = 0;
		for(i = 1; i <= 10; i++) {
			a = i * a; 
			sum = sum + a;
		}
		System.out.println("1!+2!+…+10! = " + sum);
	}

}
