win10关闭防火墙脚本.bat

@echo off
%1 mshta vbscript:CreateObject("Shell.Application").ShellExecute("cmd.exe","/c %~s0 ::","","runas",1)(window.close)&&exit
cd /d "%~dp0"
echo  关闭防火墙
NetSh Advfirewall set allprofiles state off  
echo  防火墙状态
Netsh Advfirewall show allprofiles 
pause
