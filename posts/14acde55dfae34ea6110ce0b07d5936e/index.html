#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,m,cnt=1;//cnt标志
    string str;
    cin>>n>>m;//输入
    getchar();//吸收回车
    while(n--)//
    {
        getline(cin,str);//输入字符串
        if(str.find("qiandao")!=-1||str.find("easy")!=-1);//如果符合则无输出
        else
        {
            if(m==0)//剩余做题数==0
            {
                cout<<str;//输出字符串
                cnt=0;//改标志
                break;//退出循环
            }
            m--;//做题数
        }
    }

    if(cnt)
        cout<<"Wo AK le"<<endl;//如果标志未改变

    return 0;
}
