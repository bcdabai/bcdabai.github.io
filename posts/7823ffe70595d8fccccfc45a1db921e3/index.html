c++：指向学生类的指针：求最高成绩

#include <iostream>


using namespace std;


class Student
{
public:
    Student(int n,double s)
    {
        num=n;
        score=s;
    }
void out_put(Student stu[],int i)
{


    cout<<"第"<<i+1<<"个学生学号为："<<stu[i].num<<"  学生成绩为："<<stu[i].score<<endl;


}
int getNumber()
{
    return num;
}
double getScore()
{
    return score;
}
private:
    int num;
    double score;
};
int Max(Student *A)
{
    int k;
    double MaxScore=A[0].getScore();
    for(int i=1;i<5;i++)
    {
        if(A[i].getScore()>MaxScore)                  
        {
            MaxScore=A[i].getScore();
            k=i;
        }
    }
    return A[k].getNumber();
}
int main()
{
    Student stu[5]=
    {
        Student(1001,86.5),Student(1002,85.5),Student(1003,100),
        Student(1004,98.5),Student(1005,95.5)
    };
    for(int i=0;i<5;i=i+2)
    stu[i].out_put(stu,i);
    cout<<"5个学生中成绩最高的学号为:"<<Max(stu)<<endl;
    return 0;

}
