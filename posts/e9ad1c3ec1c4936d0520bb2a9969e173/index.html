#include<bits/stdc++.h>
using namespace std;
int main(){
  int a,b,sum=0,flo=1,d=0;
    int i,m;
    for(i=0; i<12; i++) 
    {
        cin>>a;
        sum=sum+300-a;
 
        if(sum<0&&flo)
        {
            b=i+1;
            flo=0;
        }
        m=sum/100;
        sum=sum-m*100;
        d=d+m*100;
    }
    if(flo==0)	cout<<-1*b<<endl;
    else   cout<<sum+(d*6)/5<<endl;
    return 0;
}
