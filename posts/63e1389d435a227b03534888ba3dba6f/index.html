echarts缩放

echarts图表 加上dataZoom可实现缩放

 dataZoom: [//加上dataZoom可实现缩放
                {
                    type: 'slider',//x轴
                    xAxisIndex: 0,
                   
                },
                {
                    type: 'inside',//x轴
                    xAxisIndex: 0,
                    zoomOnMouseWheel:'alt',//如何触发缩放。可选值为：true：表示不按任何功能键，鼠标滚轮能触发缩放。false：表示鼠标滚轮不能触发缩放。'shift'：表示按住 shift 和鼠标滚轮能触发缩放。'ctrl'：表示按住 ctrl 和鼠标滚轮能触发缩放。'alt'：表示按住 alt 和鼠标滚轮能触发缩放。

完整代码

<!DOCTYPE html>
<html>
<head>
	<meta charset="utf-8">
	<title>echart</title>
</head>
<body>
	<!-- 为 ECharts 准备一个具备大小（宽高）的 DOM -->
<div id="main" style="width: 600px;height:400px;"></div>
</body>
</html>
<script src="https://cdn.bootcss.com/echarts/4.2.1-rc1/echarts.min.js"></script>
<script>


        // 基于准备好的dom，初始化echarts实例
        var myChart = echarts.init(document.getElementById('main'));

       option = {//折线图
            title: {
                text: "title",
            },
            tooltip: {
                trigger: 'axis',
                // formatter: formatter //百分比
            },
            toolbox: {
                feature: {
                    saveAsImage: {}//保存图片
                },
                x: 'center',
                y: 'center'
            },
            xAxis: {
                type: 'value',
                // boundaryGap: false,
                // data: dateDataArr,
                // name: '时间'
            },
            yAxis: {
                type: 'value',
                max: 'dataMax',//根据缩放改变最大值
                min: 'dataMin', // 最小值	
                // axisLabel
            },
            dataZoom: [//加上dataZoom可实现缩放
                {
                    type: 'slider',//x轴
                    xAxisIndex: 0,
                   
                },
                {
                    type: 'inside',//x轴
                    xAxisIndex: 0,
                    zoomOnMouseWheel:'alt',//如何触发缩放。可选值为：true：表示不按任何功能键，鼠标滚轮能触发缩放。false：表示鼠标滚轮不能触发缩放。'shift'：表示按住 shift 和鼠标滚轮能触发缩放。'ctrl'：表示按住 ctrl 和鼠标滚轮能触发缩放。'alt'：表示按住 alt 和鼠标滚轮能触发缩放。
                },
                // {
                //     type: 'slider',//y轴
                //     yAxisIndex: 0,
                // },
                // {
                //     type: 'inside',//y轴
                //     yAxisIndex: 0,
                // }
            ],
            series: [
                {
                    type: 'line', // 折线图
                    data: [["4.345","1.345","0.55"],["19.34","3.998","0.44"],
                        ["9.33","1.54","9.234"],["0.342","5.634","2.45"],
                        ["11.234","54.345","5.645"],["0.924","4.92","7.32"],
                        ["2.33","5.55","6.349"],["1.234","5.345","2.754"],
                        ["2.09","0.76","4.66"],["7.456","5.042","0.368"]]
                }
            ]
        };
        myChart.setOption(option);
 </script>
