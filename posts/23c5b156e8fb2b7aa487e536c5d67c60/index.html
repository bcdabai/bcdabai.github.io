循环生成的el-upload如何传递index参数

1 代码

通过.bind

<el-row v-for="(item, index) in buildSiteWorkForm.work" :key="index">
    <el-col :span="3">
      <el-form-item class="wrap" label="设备清单" prop="officeEquipmentFile">
        <el-upload
           class="upload-demo"
           action=""
           :auto-upload="false"
           ref="officeEquipmentFileRef"
           accept = ".pdf, .PDF"
           :on-change="officeEquipmentFileChangeHandle.bind(null, index)"
           :on-remove="officeEquipmentFileRemoveHandle.bind(null, index)">
           <el-button size="mini" type="primary">上传</el-button>
         </el-upload>
       </el-form-item>
    </el-col>
</el-row>
