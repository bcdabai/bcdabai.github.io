14.void sort( int a[], int n )//冒泡排序
{
    for(int i=n-1;i>0;i--)
    {
        //一趟冒泡排序
        for(int j=0;j<i;j++)
        {
            if(a[j]>a[j+1])
            {
                int t=a[j];
                a[j]=a[j+1];
                a[j+1]=t;
            }
        }
    }
}

16. 找数组元素最值int FindArrayMax( int a[], int n )// Array数组
{
    int m=0;
    for(int i=0;i<n;i++)//i为0为1均可
    {
        if(a[m]<a[i]) m=i;
    }
    return a[m];
    
    
}
