++i 和 i++的区别

/**
 * ++i 和 i++的区别
 * ++i是先加1，再用
 * i++是先用，用完再加1
 */
public class Test {
    @org.junit.Test
    public void test01(){
        int i=1;
        System.out.println(i++);// 结果是1
        System.out.println(++i);// 结果是3
        System.out.println(i);// 结果是3
    }
    @org.junit.Test
    public void test02(){
        int i=1;
        System.out.println(++i);// 结果是2
        System.out.println(i++);// 结果是2
        System.out.println(i);// 结果是3
    }

}
