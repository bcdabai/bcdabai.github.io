B：贪心，选出最长的会议，它所有的前置会议在他之前，并标记此时已经开的会议个数，时间为value+num-1，再次选择没开的会议中的最长的会议，以此类推。

#include<iostream>
#include<vector>
#include<cmath>
#include<algorithm>
using namespace std;
struct xd {
	int num, value, id;
	bool flag = 0;
	vector<int>q;
};
xd a[400005], b[400005];
bool cmp(xd a, xd b)
{
	if (a.value == b.value)
		return a.num > b.num;
	return a.value > b.value;
}
int ans = 0, tot = 0;
void dfs(int x)
{
	//int x = a[i].id;
	if (b[x].flag == 1)
		return;
	b[x].flag = 1;
	tot++;
	while (!b[x].q.empty())
	{
		dfs(b[x].q.back());
		b[x].q.pop_back();
		
	}
	return;
}
int main()
{
	int n, x;
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		cin >> a[i].value >> a[i].num;
		b[i].value = a[i].value;
		b[i].num = a[i].num;
		b[i].id = i;
		a[i].id = i;
		for (int j = 1; j <= a[i].num; j++)
		{
			cin >> x;
			a[i].q.push_back(x);
			b[i].q.push_back(x);
		}
	}
	sort(a + 1, a + 1 + n, cmp);
	
	for (int i = 1; i <= n; i++)
	{
		if (b[a[i].id].flag == 0)
		{
			dfs(a[i].id);
			ans = max(ans, tot + a[i].value - 1);
		}
	}
		cout << ans << "\n";
}
