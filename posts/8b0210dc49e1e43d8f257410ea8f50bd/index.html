php使用grpc方式调用golang服务端

proto协议

syntax = "proto3";
package service;
option go_package = "hello_grpc";
service Greeter {
  rpc SayHello (HelloRequest) returns (HelloReply) {}
}
message HelloRequest {
  string name = 1;
}
message HelloReply {
  string message = 1;
}

go版本文件生成

进入目录到 learn001中

protoc -I pb/ pb/hello_grpc.proto --go_out=plugins=grpc:service
  • -I 后面指定proto文件存放目录，和proto文件
  • –_out 这里为目标语言，这里为go
  • service 为 生成的目录，以上截图可以看到最终生成的service下存在一个.go文件

go服务端

package main
import (
	"context"
	"fmt"
	"google.golang.org/grpc"
	"google.golang.org/grpc/reflection"
	service "learn001/service"
	"net"
)
type server struct{}
func (s *server) SayHello(ctx context.Context, in *service.HelloRequest) (*service.HelloReply, error) {
	return &service.HelloReply{Message: "hello " + in.Name}, nil
}
func main() {
	// 监听本地端口
	lis, err := net.Listen("tcp", ":8080")
	if err != nil {
		fmt.Printf("监听端口失败: %s", err)
		return
	}
	// 创建gRPC服务器
	s := grpc.NewServer()
	// 注册服务
	service.RegisterGreeterServer(s, &server{})
	reflection.Register(s)
	err = s.Serve(lis)
	if err != nil {
		fmt.Printf("开启服务失败: %s", err)
		return
	}
}

启动运行main函数即可

go客户端

package main

import (
	"context"
	"fmt"
	"google.golang.org/grpc"
	service "learn001/service"
)

func main() {
	conn, err := grpc.Dial(":8080", grpc.WithInsecure())
	if err != nil {
		fmt.Println("链接失败")
		return
	}
	defer conn.Close()
	//新建一个客户端
	c := service.NewGreeterClient(conn)
	//调用服务端函数
	r, err :=c.SayHello(context.Background(), &service.HelloRequest{Name: "老骆驼"})
	if err!=nil{
		fmt.Printf("调用服务端代码失败: %s", err)
		return
	}
	fmt.Printf("调用成功: %s", r.Message)
}

php客户端

生成php版本相关文件

protoc -I pb/ pb/hello_grpc.proto --php_out=plugins=grpc:php_service

指定生成的php版本的在php_service，默认只有这两个文件夹

安装grpc扩展，windows下载dll 打开扩展即可。

composer安装grpc依赖库

新建composer.json文件内容如下：

{
  "name": "grpc-go-php",
  "require": {
    "grpc/grpc": "^v1.3.0",
    "google/protobuf": "^v3.3.0"
  },
  "autoload":{
    "psr-4":{
      "GPBMetadata\\":"GPBMetadata/",
      "Service\\":"Service/"
    }
  }
}

    'credentials' => Grpc\ChannelCredentials::createInsecure()
]);
 
//实例化 TestRequest 请求类

$request->setName("double_luo");
 
//调用远程服务
