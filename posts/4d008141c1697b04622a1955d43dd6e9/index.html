QT lineEdit控件限制输入内容（自留 仅包含部分功能后续可能添加）

第一步.h包含头文件并在private下定义指针
#include <QRegExpValidator>

 第二步.cpp文件中声明并设置限制参数

MainWindow::MainWindow(QWidget *parent) :
    QMainWindow(parent),
    ui(new Ui::MainWindow)
{
    ui->setupUi(this);

//（"[0-4]"）表示输入0-4的值（"0,1"）表示只能输入0或1

    class QRegExp regExp1("[0-4]");
    ui->lineEdit_rate->setValidator(new QRegExpValidator(regExp1,this));

//默认输入第一位，即0-4的0

    ui->lineEdit_rate->setText("0");

}
