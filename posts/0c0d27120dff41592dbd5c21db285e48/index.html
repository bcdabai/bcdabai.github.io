#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,a,i,s;
cin>>n;
cin>>a;
s=a;
for(i=2;i<=n;i++)
{
cin>>a;
if (s<a) s=a;
}
cout<<s;
	return 0;
}
