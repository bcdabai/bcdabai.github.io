C语言入门数字特征值

#include<stdio.h>
int main()
{
	int a,b,c,d=1,e=0;
	printf("输入一个0，100000的数\n");
	scanf("%d",&a);
	while(a!=0)
		{
		b=a%10;
		c++;
		a=a/10;
		if((b+c)%2==0)
		{
			e=e+d;
		}
		d=d*2;	
		}
	printf("%d",e);
	return 0;
}
