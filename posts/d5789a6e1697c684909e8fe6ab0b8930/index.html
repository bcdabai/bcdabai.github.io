将一个字符串插入另外一个字符串指定的位置

这个是把第n位挤到后一位

#include<stdio.h>
#include<string.h>
#include<stdlib.h>
 char  *change(char s1[],char s2[],int n)
 {
    int len1=strlen(s1);
    int len2=strlen(s2);
    int i;
    int j=0;
    int k=0;
    for(i=len1-1;i>=n-1;i--)  //因为插入多少长度的字符串总长度就会增加多少长度
	 {									//所以先补齐后面的字符元素先
	 	  s1[i+len2]=s1[i];  
	  } 
	 for(i=0;i<len2;i++)		//在第n位的上面插入字符串，而第n为后面接上
	   s1[n-1+i]=s2[i]; 
     return s1;
 } 
 int main()
 {
 	char s1[]="abcd" ;
	char s2[]= "qwe";
	int n=3;			//第三位
	char s3[30];
	char *newstr = s3;

	newstr = change(s1, s2, n);
	printf("%s\n", newstr);
 
 }
