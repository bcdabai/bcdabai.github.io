批量操作多个git仓库shell脚本参考

#!/bin/bash

id=""
if test -f ".id"
then
    id=$(cat .id)
else
    while [[ ! $id =~ [0-9]{8} ]]
    do
        read -p "Plz input hr id: " id
    done
    echo $id > ".id"
fi

MAIN_PROJ=ssh://${id}@gerrit.xxx.com.cn:29418/xxx（远端地址）
SUB_PROJ=(proj1, proj2, proj3)

CMD=$1

for PROJ in ${SUB_PROJ[@]}; do
        echo "========${PROJ}========"
        case $CMD in
                clone) git clone ${MAIN_PROJ}/${PROJ}
                ;;
                *) (cd $PROJ && git $*)
                ;;
        esac
done
