// 利用泰勒级数e=1+1/1！+1/2！…+1/n!计算e的近似值

二、代码


#include <stdio.h>

int main(int argc, const char * argv[]) {
    // 利用泰勒级数e=1+1/1！+1/2！....+1/n!计算e的近似值
    int i;
    float a=1,sum=1;
    for (i=1; a>=1e-5; i++) {
        a/=i;
        sum+=a;
    }
    printf("e=%f n=%d\n",sum,i+1);
    return 0;
}
