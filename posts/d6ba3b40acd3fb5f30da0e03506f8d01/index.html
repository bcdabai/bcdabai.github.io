MAC地址格式转换

格式"aa.bb.cc.dd.ee.ff"转换为""aabb-ccdd-eeff

sed -r 's#(.*)\.(.*)\.(.*)\.(.*)\.(.*)\.(.*)#\1\2-\3\4-\5\6#g' mac.txt 
OR
awk -F '.' '{print $1$2"-"$3$4"-"$5$6}' mac.txt
