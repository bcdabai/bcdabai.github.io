python画只皮卡丘

今天没事敲了一下python发现好久都没用过海龟画图了,今个画了一只简单版的皮卡丘

#接下来请欣赏像素风皮卡丘
import turtle as a
def wz(x,y):#设置一个以海龟为原点中心位移的地址
    a.penup()
    a.seth(0)
    a.fd(x)
    a.left(90)
    a.fd(y)
    a.seth(0)
    a.pendown()

def cfx(x,y):#用海龟画一个长方形，x为长，y为宽
    a.seth(0)
    a.fillcolor("black")
    a.begin_fill()
    for m in range(0, 2, 1):
        a.fd(x)
        a.right(90)
        a.fd(y)
        a.right(90)
    a.end_fill()

a.speed(0)
a.setup(900,700,400,120)#设置窗口大小
a.bgcolor("yellow")#背景黄色
