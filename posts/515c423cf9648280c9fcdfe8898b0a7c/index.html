C语言--函数_四个数找最大值

//输入四个数a,b,c,d，函数方式找出最大值。

#include <stdio.h>
int getMaxFromTwo(int x,int y)
{
	int f;
	f = x>y?x:y;
	return f;
}
int getMaxFromFour(int a1,int b1,int c1,int d1)
{
	int max1;
	max1 = getMaxFromTwo(a1,b1);
	max1 = getMaxFromTwo(max1,c1);
	max1 = getMaxFromTwo(max1,d1);
	return max1;
}

int main()
{
	int a,b,c,d;
	int max;
	
	printf("请输入四个整数：");
	scanf("%d%d%d%d",&a,&b,&c,&d);
	max = getMaxFromFour(a,b,c,d);
	printf("最大数为：%d",max);
	
	return 0;
}
