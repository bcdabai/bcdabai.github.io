#springboot+jdbc连接数据库
spring:
  datasource:
    type: com.alibaba.druid.pool.DruidDataSource #指定使用阿里的连接池
    druid:
    driver-class-name: com.mysql.cj.jdbc.Driver
    url: jdbc:mysql://localhost:3306/royal?useSSL=false&autoReconnect=true&useUnicode=true&characterEncoding=UTF8&zeroDateTimeBehavior=convertToNull&allowMultiQueries=true
    username: root
    password: 0429

现在的配置文件

解决：只需要在密码前后加上引号，就可以保证密码的字符特性，不被识别成数字而改变。

#springboot+jdbc连接数据库
spring:
  datasource:
    type: com.alibaba.druid.pool.DruidDataSource #指定使用阿里的连接池
    druid:
    driver-class-name: com.mysql.cj.jdbc.Driver
    url: jdbc:mysql://localhost:3306/royal?useSSL=false&autoReconnect=true&useUnicode=true&characterEncoding=UTF8&zeroDateTimeBehavior=convertToNull&allowMultiQueries=true
    username: root
    password: "0429"

在Springboot中pom.xml文件连接数据库导入的jar包

<dependency>
            <groupId>mysql</groupId>
            <artifactId>mysql-connector-java</artifactId>
            <scope>runtime</scope>
        </dependency>
        <dependency>
            <groupId>com.alibaba</groupId>
            <artifactId>druid-spring-boot-starter</artifactId>
            <version>1.2.8</version>
        </dependency>
