cmd 字符串替换

@echo off
REM 字符串替换

set str=待替换文本信息XXX
echo 对“%str%”中的“XXX”进行替换，替换为“YYY”

REM 替换str中的XXX为YYY
set str2=%str:XXX=YYY%

echo -
echo 替换后为：%str2%
echo -
pause
