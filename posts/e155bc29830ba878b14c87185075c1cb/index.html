luvcview分析--1.环境搭建

一. 源码编译及目标

  2. cong@msi:/tmp$ sudo apt-get source luvcview //b. 这就在当前目录下
  3. cong@msi:/tmp$ ls

3. libv4l库的编译
  1. libv4l的库源码下载地址:
  2. http://people.atrpms.net/~hdegoede/

  4. cong@msi:/work/libv4l2/libv4l-0.6.1$ ls
  5. include libv4l2 libv4lconvert Makefile README

  7. cong@msi:/work/libv4l2/libv4l-0.6.1$ make LINKTYPE=static //直接make就可以了，这儿是编成静态库
4. libv4l库的使用
  1. Makefile中:
  2. CFLAGS=-g -O0 -I/work/test/uvcview/libv4l2/libv4l-0.6.1/include               //添加include路径
  3. V4L2LIBS = -L/work/test/uvcview/libv4l2/libv4l-0.6.1/libv4l2 -lv4l2 \         //添加库的链接路径
  4.            -L/work/test/uvcview/libv4l2/libv4l-0.6.1/libv4lconvert/ -lv4lconvert -lrt -lm
注意:
a. 出现undefined reference to `shm_open',是因为libv4l2库中用到了,需要在编译时加上-lrt
b. 出现undefined reference to `powf'，是因为libv4l2库中用到了,需要在编译时加上-lm
c. 为了防止系统中己安装的v4l2库的干扰,把v4l2删掉: sudo apt-get remove libv4l-dev  
5. 目标
目标是通过分析luvcview,自己动手写一个luvcview,这不是再次造轮子,而是一个学习并提高的过程。
二. 测试demo
2.1 check
获取camera所支持的视频格式
  1. cong@msi:/work/test/uvcview/1check$ cat check.c 
  2. #include "utils.h"
  3. #include <sys/types.h>
  4. #include <sys/stat.h>
  5. #include <fcntl.h>
  6. #include <linux/videodev2.h>

  8. #define VIDEO_DEVICE "/dev/video0"
  9. int main ( int argc, char *argv[] )
  10. {
  11.     int fd;
  12.     int ret = 0;
  13.     struct v4l2_capability cap;
  14.     struct v4l2_fmtdesc fmt;
  15.     if( (fd = v4l2_open(VIDEO_DEVICE,O_RDWR)) == -1)       //打开设备
  16.     {
  17.         dbmsg("error: %s", strerror(errno));
  18.         return 0;
  19.     }
  20.     memset(&cap, 0, sizeof(struct v4l2_capability));        
  21.     if( (ret = v4l2_ioctl(fd, VIDIOC_QUERYCAP, &cap) < 0))  //查询设备属性
  22.     {
  23.         dbmsg("error: %s", strerror(errno));
  24.         return 0;
  25.     }
  26.     //接下来就可以进行一系列的判断了
  27.     //cap.capabilities & V4L2_CAP_VIDEO_CAPTURE    //判断是否支持图像获取
  28.     //cap.capabilities & V4L2_CAP_STREAMING        //是否具有数据流控制模式
  29.     //cap.capabilities & V4L2_CAP_READWRITE        //是否支持 read和write I/O操作函数
  30.     memset(&fmt, 0, sizeof(struct v4l2_fmtdesc));
  31.     fmt.index = 0; 
  32.     fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
  33.     while ((ret = v4l2_ioctl(fd, VIDIOC_ENUM_FMT, &fmt)) == 0) {    //获取当前视频设备支持的视频格式
  34.             printf("{ pixelformat = '%c%c%c%c', description = '%s' }\n",
  35.                     fmt.pixelformat & 0xFF, (fmt.pixelformat >> 8) & 0xFF,
  36.                     (fmt.pixelformat >> 16) & 0xFF, (fmt.pixelformat >> 24) & 0xFF,
  37.                     fmt.description);
  38.             if(ret != 0)
  39.                 printf(" Unable to enumerate frame sizes.\n");
  40.         fmt.index++;
  41.     }
  42.     return EXIT_SUCCESS;
  43. }
b. Makefile
  1. cong@msi:/work/test/uvcview/1check$ cat Makefile 
  2. EXE=check
  3. CC=gcc
  4. SRC=$(wildcard *.c)
  5. #OBJ=$(SRC:.c=.o)
  6. OBJ=$(patsubst %.c,%.o,$(SRC))
  7. DEP=$(patsubst %.c,.%.d,$(SRC))
  8. CFLAGS=-g -O0 -I/work/test/uvcview/libv4l2/libv4l-0.6.1/include
  9. #V4L2LIBS = $(shell pkg-config --libs libv4l2)
  10. V4L2LIBS = -L/work/test/uvcview/libv4l2/libv4l-0.6.1/libv4l2 -lv4l2 \
  11.          -L/work/test/uvcview/libv4l2/libv4l-0.6.1/libv4lconvert/ -lv4lconvert \
  12.          -lrt -lm
  13. $(EXE):$(OBJ)
  14.     $(CC) $(CFLAGS) $^ -o $@ $(V4L2LIBS) 

  16. $(DEP):.%.d:%.c
  17.     @set -e; rm -f $@; \
  18.     $(CC) -MM $< > $@.$$$$; \
  19.     sed 's,/($*/)/.o[ :]*,/1.o $@ : ,g' < $@.$$$$ > $@; \
  20.     rm -f $@.$$$$

  22. -include $(DEP)
  23. clean:
  24.     @rm $(EXE) $(OBJ) $(DEP) -f
