Nginx proxy_pass 转发规则配置

/api1/ >> http://ip:port

http://localhost/api1/xxx >> http://localhost:8080/api1/xxx

server {
    listen 80;
    location /api1/ {
        proxy_pass http://localhost:8080;
    }
}

/api1/ >> http://ip:port/
