C语言，输入一串字符，统计有多少个字母

#include<stdio.h>
int main()
{
    char a;
    int num=0;
    while((a=getchar())&&a!='\n')
    {
        if((a>='a'&&a<='z')||(a>='A'&&a<='Z'))
        num++;
    }
    printf("%d\n",num);
    return 0;
}
