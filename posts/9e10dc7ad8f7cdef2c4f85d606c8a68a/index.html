Linux下配置java环境版本opencv

#安装gcc等编译环境
yum install gcc
yum install python3.8
yum install cmake
yum groupinstall "Development Tools"
#去官方网站下载使用的压缩包4.5.0版本
#在local目录下创建一个opencv文件夹
mkdir /usr/local/opencv
#解压opencv安装包
tar -zxvf opencv-4.5.0.tar.gz
mkdir build
cd build
cmake -D CMAKE_BUILD_TYPE=RELEASE -D CMAKE_INSTALL_PREFIX=/usr/local ..
make 
make install
安装自动化编译工具ant
yum install ant
cmake -D CMAKE_BUILD_TYPE=RELEASE -D CMAKE_INSTALL_PREFIX=/usr/local -DBUILD_TESTS=OFF .. 
make -j8
make install
