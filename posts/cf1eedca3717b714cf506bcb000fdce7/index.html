有26个字母a~z，找出所有字母组合，a、b、c、ab、abc、a~z 都是一个组合(顺序无关)

mark 一下，好像是用深搜做的，目前看不太懂。

int list[30];
int ans = 0;
void work(int pnt, int l, int now)
{
	int i;
	if (now > l)
		ans++;
	else
		for (i = pnt; i <= 26; i++)
			work(i + 1, l, now + 1);
}

int full_combination()
{
	int i;
	for (i = 1; i <= 26; i++)
		work(1, i, 1);
	return ans;
}
