# 摩斯码转字符串
def decipherForMos():
    a =input("input the string:")
    s = a.split(" ") #按空格将输入的字符切片
    for item in s:
        print(dict[item],end='')
# 字符串转摩斯码
def decipherForStr():
    a = input("input the string:")
    s =list(a)
    for t in s:
        for k,v in dict.items():
            if v==t:
                print(k,end=' ')


if __name__ == '__main__':
    decipherForStr()
