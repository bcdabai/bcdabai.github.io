用切片操作去除字符串首尾的指定字符串

发现自己的抽象能力和逻辑能力好像并不好，这个代码的逻辑演练了好久

def trim(text, s=' '):
    """ 利用切片实现去除首尾指定字符串 """

    for i in range(2):
    
        temp = ''
        text = text[::-1] if i == 1 else text
    
        for index, i in enumerate(text):
            temp += i
    
            if len(temp) == len(s):
                if temp != s:
                    text = text[index-(len(s)-1):]
                    break
                temp = ''
                
        if temp == '':
            return temp
        
        s = s[::-1]
    
    text = text[::-1]
    
    return text


print(trim('nbnbpytonnbnb', 'nb'))
