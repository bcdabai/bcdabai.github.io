def getACC(Y_test,Y_pred,n):
    acc =[]
    con_mat = confusion_matrix(Y_test,Y_pred)
    for i in range(n):
        number = np.sum(con_mat[:,:])
        tp = con_mat[i][i]
        fn = np.sum(con_mat[i,:])- tp
        fp = np.sum(con_mat[:,i])- tp
        tn = number - tp - fn - fp
        acc1 =(tp+tn)/(number)
        acc.append(acc1)
    return acc
