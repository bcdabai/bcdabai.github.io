用js，编写一个函数，计算三个数字的大小，按从小到大的顺序输出

 // 第一种方法
        function compare(num1, num2, num3) {
            if (num1 > num2) {
                if (num2 > num3) {
                    console.log('从小到大的顺序：' + num3 + '>>' + num2 + '>>' + num1);
                    // num3>num2 
                } else {
                    console.log('从小到大的顺序：' + num2 + '>>' + num3 + '>>' + num1);
                }
                //num1<num2
            } else {
                if (num2 < num3) {
                    console.log('从小到大的顺序：' + num1 + '>>' + num2 + '>>' + num3);
                    // num3<num2
                } else {
                    console.log('从小到大的顺序：' + num1 + '>>' + num3 + '>>' + num2);
                }
            }
        }
        compare(10, 2, 6);

 // 第二种方法
        function compare(num1, num2, num3) {
        // 只要case表达式的值， 等于switch的内容 就能进入该case
            switch (true) {
                // num1为最小情况
                case num1 < num2 && num2 < num3:
                    return [num1, num2, num3]
                case num1 < num3 && num3 < num2:
                    return [num1, num3, num2]
                    // num2为最小情况
                case num2 < num1 && num1 < num3:
                    return [num2, num1, num3]
                case num2 < num3 && num3 < num1:
                    return [num2, num3, num1]
                    // num3为最小情况
                case num3 < num1 && num1 < num2:
                    return [num3, num1, num2]
                case num3 < num2 && num2 < num1:
                    return [num3, num2, num1]

                default:
                    console.log('鬼打墙');
                    break;
            }
        }
        console.log(compare(3, 5, 4));
