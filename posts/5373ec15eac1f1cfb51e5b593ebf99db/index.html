#include <iostream>
using namespace std;
class A
{
public:
    A(){cout<<"A";}
    ~A(){cout<<"~A";}
};

class B
{
    A *p;
public:
    B()
    {
        cout<<"B";
        p=new A();
    }
    ~B()
    {
        cout<<"~B";
        delete p;
    }
};
int main()
{
    B obj;
    return 0;
}


运行结果：

         BA~B~A

B：声明B类的“obj”执行B类的构造函数，输出“B”

A：在执行B的构造函数时，给A类型的*p分配动态内存，从而执行A类的构造函数，输出“A”

~B：程序运行结束，释放obj的内存，输出“~B”

~A：在执行B类的虚构函数时执行A的虚构函数，输出“~A”
