def add_up(data):               # ç´¯åŠ å‡½æ•°
    sum = 0
    for i in range(len(data)):
        sum += data[i]
    return sum

def cal_a(x,y):                 # æ±‚a
    a = (len(x) * add_up(x * y) - add_up(x) * add_up(y)) / (len(x) * add_up(x ** 2) - add_up(x) ** 2)
    return a

def cal_b(x,y,a):               # æ±‚b
    b = add_up(y) / len(x) - a * add_up(x) / len(x)
    return b
