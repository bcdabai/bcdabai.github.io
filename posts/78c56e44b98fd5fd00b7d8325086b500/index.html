def hadn(m):
    had, n = {"瓶子": m, "盖子": m}, m
    while had["瓶子"] >= 2 or had["盖子"] >= 4:
        a, b = had["瓶子"] // 2, had["盖子"] // 4
        had["瓶子"], had["盖子"], n = had["瓶子"] - a + b, had["盖子"] + a - 3 * b, n + a + b
    else:
        print(n)
