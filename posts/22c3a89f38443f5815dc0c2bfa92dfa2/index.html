启动jar包报错: 找不到或无法加载主类

<plugin>
                <groupId>org.apache.maven.plugins</groupId>
                <artifactId>maven-jar-plugin</artifactId>
                <configuration>
                    <!--不打包资源文件-->
                    <excludes>
                        <exclude>static/**</exclude>
                        <!--<exclude>db/**</exclude>-->
                        <exclude>*.txt</exclude>
                        <exclude>*.xml</exclude>
                        <exclude>*.yml</exclude>
                    </excludes>
                    <archive>
                        <manifest>
                            <!-- 是否要把第三方jar加入到类构建路径 -->
                            <addClasspath>true</addClasspath>
                            <!-- 外部依赖jar包的最终位置 -->
                            <classpathPrefix>lib/</classpathPrefix>
                            <!--指定jar程序入口-->
                            <mainClass>com.attanzai.wms.WmsApplication</mainClass>
                            <!--SNAPSHOT版本问题-->
                            <useUniqueVersions>false</useUniqueVersions>
                        </manifest>
                        <!--MANIFEST.MF 中 Class-Path 加入资源文件目录-->
                        <manifestEntries>
                            <Class-Path>./config/</Class-Path>
                        </manifestEntries>
                    </archive>
                </configuration>
            </plugin>
