//预览
       function preview(ID){
            var murl = '<%=basePath%>wxwebindex/previewcourseware.do?ID='+ID;
             if(ID=='' || ID==null){
                alert("请先保存信息!");
               }
               else{        
            $("#preview-html").attr('src', murl);
            $("#zhongxin").hide();
            $(".preview-layer").css('display','block');
              }  
        }
               //关闭预览
       function previewclose(){
          $(".preview-layer").css('display','none');
          $("#zhongxin").show();         
           }
