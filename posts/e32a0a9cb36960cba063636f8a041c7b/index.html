使用c语言来完成一个学生管理系统，主要功能有，姓名根据成绩排序，以及查找学生信息，计算总分、平均分

#include <stdio.h>
#include <string.h>

// 定义学生结构体
struct Student {
    char name[50];
    int grade;
    int age;
    char gender[40];
    float score;
};

// 函数声明
void findStudent(struct Student students[], int numStudents, const char* searchName);
float calculateTotalScore(struct Student student);
float calculateAverageScore(struct Student student);
float calculateSubjectTotal(struct Student students[], int numStudents);
float calculateSubjectAverage(struct Student students[], int numStudents);

int main() {
    // 动态输入学生人数
    setbuf(stdout, NULL);

    int numStudents;
    printf("请输入学生人数: ");
    scanf_s("%d", &numStudents);

    // 初始化学生数组
    struct Student students[numStudents];

    // 输入学生信息
    for (int i = 0; i < numStudents; ++i) {
        printf("输入学生详情 %d:\n", i + 1);

        printf("Name: ");
        scanf("%s", students[i].name);

        printf("Grade: ");
        scanf_s("%d", &students[i].grade);

        printf("Age: ");
        scanf_s("%d", &students[i].age);

        printf("Gender (男/女): ");
        scanf(" %s", students[i].gender);

        printf("Score: ");
        scanf_s("%f", &students[i].score);

        printf("\n");
    }

    // 计算总分、平均分、最高分和最低分
    float sum = 0;
    float maxScore = students[0].score;
    float minScore = students[0].score;

    for (int i = 0; i < numStudents; ++i) {
        sum += students[i].score;

        if (students[i].score > maxScore) {
            maxScore = students[i].score;
        }

        if (students[i].score < minScore) {
            minScore = students[i].score;
        }
    }

    // 计算平均分
    float average = sum / numStudents;

    // 打印每个学生的信息
    printf("\n学生信息:\n");
    for (int i = 0; i < numStudents; ++i) {
        printf("Student %d:\n", i + 1);
        printf("Name: %s\n", students[i].name);
        printf("Grade: %d\n", students[i].grade);
        printf("Age: %d\n", students[i].age);
        printf("Gender: %s\n", students[i].gender);
        printf("Score: %.2f\n", students[i].score);
        printf("\n");
    }

    // 打印计算结果
    printf("结果:\n");
    printf("总分: %.2f\n", sum);
    printf("平均分: %.2f\n", average);
    printf("最高分: %.2f\n", maxScore);
    printf("最低分: %.2f\n", minScore);

    // 查找学生
    char searchName[50];
    printf("\n输入要查找的学生姓名: ");
    scanf("%s", searchName);
    findStudent(students, numStudents, searchName);

    // 计算每个学生的总成绩和平均分
    printf("\n每个学生的总成绩和平均分:\n");
    for (int i = 0; i < numStudents; ++i) {
        printf("Student %d:\n", i + 1);
        printf("总成绩: %.2f\n", calculateTotalScore(students[i]));
        printf("平均分: %.2f\n", calculateAverageScore(students[i]));
        printf("\n");
    }

    // 计算每门课程的总分和平均分
    printf("每门课程的总分和平均分:\n");
    float subjectTotal = calculateSubjectTotal(students, numStudents);
    float subjectAverage = calculateSubjectAverage(students, numStudents);
    printf("总分: %.2f\n", subjectTotal);
    printf("平均分: %.2f\n", subjectAverage);

    // 按成绩排序学生
    printf("\n按成绩排序后的学生信息:\n");
    for (int i = 0; i < numStudents - 1; ++i) {
        for (int j = 0; j < numStudents - i - 1; ++j) {
            if (students[j].score < students[j + 1].score) {
                // 交换位置
                struct Student temp = students[j];
                students[j] = students[j + 1];
                students[j + 1] = temp;
            }
        }
    }

    // 打印按成绩排序后的学生信息
    for (int i = 0; i < numStudents; ++i) {
        printf("Student %d:\n", i + 1);
        printf("Name: %s\n", students[i].name);
        printf("Grade: %d\n", students[i].grade);
        printf("Age: %d\n", students[i].age);
        printf("Gender: %s\n", students[i].gender);
        printf("Score: %.2f\n", students[i].score);
        printf("\n");
    }

    return 0;
}

// 查找学生
void findStudent(struct Student students[], int numStudents, const char* searchName) {
    int found = 0;
    for (int i = 0; i < numStudents; ++i) {
        if (strcmp(students[i].name, searchName) == 0) {
            found = 1;
            printf("找到学生:\n");
            printf("Name: %s\n", students[i].name);
            printf("Grade: %d\n", students[i].grade);
            printf("Age: %d\n", students[i].age);
            printf("Gender: %s\n", students[i].gender);
            printf("Score: %.2f\n", students[i].score);
            break;
        }
    }
    if (!found) {
        printf("未找到学生。\n");
    }
}

// 计算学生总成绩
float calculateTotalScore(struct Student student) {
    return student.score;
}

// 计算学生平均分
float calculateAverageScore(struct Student student) {
    return student.score;
}

// 计算每门课程的总分
float calculateSubjectTotal(struct Student students[], int numStudents) {
    float total = 0;
    for (int i = 0; i < numStudents; ++i) {
        total += students[i].score;
    }
    return total;
}

// 计算每门课程的平均分
float calculateSubjectAverage(struct Student students[], int numStudents) {
    return calculateSubjectTotal(students, numStudents) / numStudents;
}
