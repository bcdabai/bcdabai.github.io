#include<stdio.h>
int main()
{   char g;
    scanf("%c",&g);
   printf("  %c   \n",g);
   printf(" %c%c%c \n",g,g,g);
   printf("%c%c%c%c%c \n",g,g,g,g,g);
    return 0;
}
