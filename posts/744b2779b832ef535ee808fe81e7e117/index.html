#include<iostream>
#define MAX 1000
using namespace std;

//菜单
void showMenu() {
	cout << "┏━━━━━━━━━━━━━┓" << endl;
	cout << "┃ 1.添加联系人┃" << endl;
	cout << "┣━━━━━━━━━━━━━┫" << endl;
	cout << "┃ 2.显示联系人┃" << endl;
	cout << "┣━━━━━━━━━━━━━┫" << endl;
	cout << "┃ 3.删除联系人┃" << endl;
	cout << "┣━━━━━━━━━━━━━┫" << endl;
	cout << "┃ 4.查找联系人┃" << endl;
	cout << "┣━━━━━━━━━━━━━┫" << endl;
	cout << "┃ 5.修改联系人┃" << endl;
	cout << "┣━━━━━━━━━━━━━┫" << endl;
	cout << "┃ 6.清空联系人┃" << endl;
	cout << "┣━━━━━━━━━━━━━┫" << endl;
	cout << "┃ 7.退出通讯录┃" << endl;
	cout << "┗━━━━━━━━━━━━━┛" << endl;

struct Person{
	string name;
	int sex;
	int age;
	int phone;
	string address;

};
struct Addressbooks{
	Person personArray[MAX];
	int size;
};
//添加联系人
void addPerson(Addressbooks* abs) {
	if (abs->size == MAX) {
		cout << "通讯录已满" << endl;
		return;
	}
	else {
		string name;
		cout << "请输入姓名:" << endl;
		cin >> name;
		abs->personArray[abs->size].name = name;

		int sex=0;
		cout << "请输入性别:" << endl;
		cout << "1 --- 男" << "\t" << "2 --- 女" << endl;
		while(true) {
			cin >> sex;
			if (sex == 1 || sex == 2) {
				abs->personArray[abs->size].sex = sex;
				break;
				
			}
			else {
				cout << "输入有误，请重新输入" << endl;
			}
		}

		int age;
		cout << "请输入年龄:" << endl;
		cin >> age;
		abs->personArray[abs->size].age = age;

		int phone;
		cout << "请输入手机号:" << endl;
		cin >> phone;
		abs->personArray[abs->size].phone = phone;

		string address;
		cout << "请输入地址:" << endl;
		cin >> address;
		abs->personArray[abs->size].address = address;

		abs->size++;
		cout << "添加成功" << endl;
		system("pause");
		system("cls");//清屏
	}
}
//查找
void showPerson(Addressbooks* abs) {
	if (abs->size == 0) {
		cout << "当前没有添加联系人" << endl;
	}
	else {
		for (int i = 0; i < abs->size; i++) {
			cout << "姓名: " << abs->personArray[i].name << "\t";
			cout << "性别: " << (abs->personArray[i].sex == 1 ? "男" : "女") << "\t";
			cout << "年龄: " << abs->personArray[i].age << "\t";
			cout << "手机号 :" << abs->personArray[i].phone << "\t";
			cout << "地址: " << abs->personArray[i].address << endl;
		}		
	}
	system("pause");
	system("cls");
}
//检测
int checkPreson(Addressbooks* abs,string name) {
	for (int i = 0; i < abs->size; i++) {
		if (name == abs->personArray[i].name) {
			return i;
		}
	}
	return -1;
}
//删除
void deletePerson(Addressbooks* abs) {
	cout << "请输入要删除的联系人的姓名" << endl;
	string name;
	cin >> name;
	int ret = checkPreson(abs,name);
	if (ret != -1) {
		for (int i = ret; i < abs->size; i++) {
			abs->personArray[i] = abs->personArray[i + 1];
		}
		abs->size--;
		cout << "删除成功" << endl;
	}
	else{
		cout << "查无此人" << endl;
	}
	system("pause");
	system("cls");
}
//查找
void findPerson(Addressbooks* abs) {
	cout << "请输入你要查找的联系人：" << endl;
	string name;
	cin >> name;
	int ret = checkPreson(abs, name);
	if (ret != -1) {
		cout << "姓名: " << abs->personArray[ret].name << "\t";
		cout << "性别: " << (abs->personArray[ret].sex == 1 ? "男" : "女") << "\t";
		cout << "年龄: " << abs->personArray[ret].age << "\t";
		cout << "手机号 :" << abs->personArray[ret].phone << "\t";
		cout << "地址: " << abs->personArray[ret].address << endl;
	}
	else {
		cout << "查无此人" << endl;
	}
	system("pause");
	system("cls");
}
//修改
void modfyPerson(Addressbooks* abs) {
	cout << "请输入你要修改的联系人：" << endl;
	string name;
	cin >> name;
	int ret = checkPreson(abs, name);
	if (ret != -1) {
		int sex = 0;
		cout << "请输入性别:" << endl;
		cout << "1 --- 男" << "\t" << "2 --- 女" << endl;
		while (true) {
			cin >> sex;
			if (sex == 1 || sex == 2) {
				abs->personArray[ret].sex = sex;
				break;
				cout << "输入有误，请重新输入" << endl;
			}
		}

		int age;
		cout << "请输入年龄:" << endl;
		cin >> age;
		abs->personArray[ret].age = age;

		int phone;
		cout << "请输入手机号:" << endl;
		cin >> phone;
		abs->personArray[ret].phone = phone;

		string address;
		cout << "请输入地址:" << endl;
		cin >> address;
		abs->personArray[ret].address = address;

		cout << "修改成功" << endl;
		system("pause");
		system("cls");

	}
}
//清空
void clearPerson(Addressbooks* abs) {
	abs->size = 0;
	cout << "通讯录已清空成功" << endl;
	system("pause");
	system("cls");
}
int main() {
	Addressbooks abs;
	abs.size = 0;
	int select = 0;//用户输入
	while (true) {
		showMenu();

		cin >> select;
		switch (select) {
		case 1://添加
			addPerson(&abs);
			break;
		case 2://显示
			showPerson(&abs);
			break;
		case 3://删除
			deletePerson(&abs);
			break;
		case 4://查找
			findPerson(&abs);
			break;
		case 5://修改
			modfyPerson(&abs);
			break;
		case 6://清空
			clearPerson(&abs);
			break;
		case 7://退出
			cout << "欢迎下次使用哟" << endl;
			system("pause");
			return 0;
			break;
		default:
			break;
		}
	}
	system("pause");
	return 0;
}
