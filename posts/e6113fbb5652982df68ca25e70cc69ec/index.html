uniapp 获取手机号

 1. wxml

<u-button slot="right" customStyle="background: #21C9A0;border-radius: 0rpx 8rpx 8rpx 0rpx;color: #fff;" class="u-button-phone" type="info" open-type="getPhoneNumber" @getphonenumber="getPhoneNumber">获取手机号</u-button>



<u-input type="number" v-model="formData.phone" clearable placeholder="请输入手机号" border="none"></u-input>

2.js

    import {
       getPhone
    } from '@/api/index'
    methods:{
        getPhoneNumber (e) {
            console.log(e.detail.code)
            getPhone({code:e.detail.code}).then(res =>{
              // 手机号
              this.formData.phone = res.data.data.phoneInfo.purePhoneNumber;
            }).catch(err=>{
              uni.$u.toast('获取失败，请重试');
            })
        },
    }
