Spring Boot之使用指定的JDK版本

只需要在pom.xml中进行配置

<!-- JDK版本 -->
        <plugin>
		   <artifactId>maven-compiler-plugin</artifactId>
		   <configuration>
		      <source>1.7</source>
		      <target>1.7</target>
		   </configuration>
		</plugin>
