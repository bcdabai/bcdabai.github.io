C语言—阶乘的计算1！+2！+...+10！

计算1！+2！+…+10！

第一种

#include<stdio.h>
int main()
{
int i = 0;
int n = 0;
int ret = 1;
int sum = 0;
for (n = 1; n <= 10; n++)
{
ret = 1;
for (i = 1; i <= n; i++)
{
ret = ret * i; //n的阶乘
}
sum = sum + ret;
}
printf(“ret = %d\n”, sum);
return 0;
}
