输出数字金字塔(数字1-9)

#include <stdio.h>
//输出数字金字塔(数字1-9)
int main()
{
	int n,i,j,k;
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=n-i;j++) printf(" ");
		for (k=1;k<=i;k++)
		{
			if (k==i)
			{
				printf("%d",i);
			}else{
				printf("%d ",i);
			}
		}
		printf("\n");  
	}
	return 0;
}
