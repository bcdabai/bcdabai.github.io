C语言实现三子棋（嘎嘎权威）

1.介绍

对于实现三子棋游戏我使用了3个文件分别是1.头文件（game.h）放函数的声明，#include包含的头文件,以及#define定义的常量都会放到game.h这个头文件中。2.（game.c）游戏中的各种函数实现放到了其中。3.(test.h)游戏的大体框架包括游戏的入口函数，界面效果等等。接下来我们从各个文件开始了解

2.代码

game.h

#pragma once
#define ROW 3
#define COL 3
#include<stdio.h>
#include<stdlib.h>
#include<time.h>
void Init_board(char board[ROW][COL],int row,int col);//初始化棋盘
void Display_board(char board[ROW][COL],int row,int col);//打印棋盘
void playermove(char board[ROW][COL], int row, int col);//玩家下棋
void computmove(char board[ROW][COL],int row,int col);//电脑下棋
char is_win(char board[ROW][COL], int row, int col);//判断输赢
//玩家赢-'*'
//电脑赢-'#'
//平局-'Q'
//继续-'C'

上诉代码中包含了游戏实现的大部分函数的声明，在头文件声明的函数可以在不同的.c文件中依靠#include"game.h",包含头文件来使用

game.c

#include"game.h"
void Init_board(char board[ROW][COL], int row, int col)
{
	int i = 0;
	int j = 0;
	for (i = 0; i < row; i++)
	{
		for (j = 0; j < col; j++)
		{
			board[i][j] = ' ';
		}
	}
}
void Display_board(char board[ROW][COL], int row, int col)
{
	int i = 0;
	int j = 0;
	for (i = 0; i < row; i++)
	{
		for (j = 0; j < col; j++)
		{
			printf(" %c ", board[i][j]);
			if (j < col - 1)
			{
				printf("|");
			}
		}
		printf("\n");

		if (i < row - 1)
		{
			for (j = 0; j < col; j++)
			{
				printf("---");
				if (j < col - 1)
				{
					printf("|");
				}
			}
			printf("\n");
		}
	}
}
void playermove(char board[ROW][COL], int row, int col)
{
	int x = 0;
	int y = 0;
	while (1)
	{
		printf("玩家下:\n");
		scanf("%d %d", &x, &y);
		if (x >= 1 && x <= row && y >= 1 && y <= col)
		{
			if (board[x - 1][y - 1] == ' ')
			{
				board[x - 1][y - 1] = '*';
				break;
			}
			else
			{
				printf("该坐标被占用，重新选择:\n");
			}
		}
		else
		{
			printf("坐标非法，请重新输入坐标:\n");
		}

	}
}
void computmove(char board[ROW][COL], int row, int col)
{
	int x = 0;
	int y = 0;
	printf("电脑下:\n");
	while (1)
	{
		x = rand() % row;
		y = rand() % col;
		if (board[x][y] == ' ')
		{
			board[x][y] = '#';
			break;
		}
	}
}
char is_full(char board[ROW][COL], int row, int col)
{
	int i = 0;
	int j = 0;
	for(i = 0; i < row; i++)
	{
		for (j = 0; j < col; j++)
		{
			if (board[i][j] == ' ')
			{
				return 0;
			}
		}
	}
	return 1;
}
char is_win(char board[ROW][COL], int row, int col)
{
	int i = 0;
	for (i = 0; i < row; i++)
	{
		if (board[i][0] == board[i][1] && board[i][1] == board[i][2] && board[i][1] != ' ')//行
		{
			return board[i][1];
		}
	}
	for (i = 0; i < col; i++)
	{
		if (board[0][i] == board[1][i] && board[1][i] == board[2][i] && board[1][i] != ' ')//列
		{
			return board[1][i];
		}
	}
	if (board[0][0] == board[1][1] && board[1][1] == board[2][2] && board[1][1] != ' ')
	{
		return board[1][1];
	}
	if (board[0][2] == board[1][1] && board[1][1] == board[2][0] && board[1][1] != ' ')
	{
		return board[1][1];
	}
	if (1 == is_full(board, row, col))
	{
		return 'Q';
	}
	return 'C';
}

在game.c中具体写出了在游戏中用到的函数实现的过程，可以结合test.c来看看那个函数的实现不懂。

test.c

#define _CRT_SECURE_NO_WARNINGS 1
#include"game.h"
void menu()
{
	printf("**********************\n");
	printf("****    1.play    ****\n");
	printf("****    0.exit    ****\n");
	printf("*********************\n");
}
void game()
{
	char board[ROW][COL] = { 0 };
	Init_board(board,ROW,COL);//初始化棋盘
	Display_board(board, ROW, COL);//打印棋盘
	//下棋
	char ret = 0;
	while (1)
	{
		playermove(board,ROW,COL);//玩家下棋
		Display_board(board, ROW, COL);
		ret=is_win(board,ROW,COL);
		if (ret != 'C')
		{
			break;
		}
		computmove(board, ROW, COL);//电脑下棋
		Display_board(board, ROW, COL);
		ret = is_win(board,ROW,COL);
		if (ret!='C')
		{
			break;
		}
	}
	if (ret == '*')
	{
		printf("玩家赢\n");
	}
	else if (ret=='#')
	{
		printf("电脑赢\n");
	}
	else
	{
		printf("平局\n");
	}
}
void test()
{
	int input=0;
	srand((unsigned int)time(NULL));
	do
	{
		menu();
		printf("请选择\n");
		scanf("%d", &input);
		switch (input)
		{
		case 1:
			game();
			break;
		case 0:
			printf("退出游戏\n");
			break;
		default:
			printf("选择错误，重新选择\n");
			break;
		}
	} while (input);
}
int main()
{
	test();
	return 0;
}

test.c中就是三子棋实现的框架了，我们为了能一直玩游戏就需要用到循环，又因为需要先打印菜单就使用了do while();循环

3.过程

有了以上的文件就可以实现三子棋游戏了，我们通过menu():选择1.玩或0.退出，选1进入了game();函数，可以将棋盘想象为是一个3*3的二维数组，开始棋盘为空格，随着玩家放置*和电脑放置#来改变二维数组中的内容，以实现双方下棋的操作。随着不同的‘棋子’放入’棋盘’中我们就需要对于游戏进行判断了，游戏的输赢1.玩家赢2.电脑赢.3平局 结局就这3种，可是什么时间判断呢，为了保证准确就需要在每一次下棋后就判断，也就出现了都没赢’继续’的选项，所以有4种可能性判断结果。也就实现了三子棋游戏，我接着回来介绍一下实现玩家，电脑下棋的思路。首先玩家下棋就是通过输入坐标实现对应的位置放置棋子，电脑下棋，电脑下棋就需要用到随机数生成的概念了，通过生成的随机数作为坐标的参数，使得实现了电脑下棋的过程，有了这些思路结合代码我相信就可以学会三子棋，有需要的朋友可以自己复制代码测试一下，加深对于代码的理解
