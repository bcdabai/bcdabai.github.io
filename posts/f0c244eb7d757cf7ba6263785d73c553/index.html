A.Split it!

题意：给你一个字符串s,和一个参数k,问是否存在k+1个非空子串 a 1 a_1 a1​, a 2 a_2 a2​,… a k a_k ak​, a k + 1 a_{k+1} ak+1​使得 s = a 1 + a 2 + . . . + a k + a k + 1 + R ( a k ) + . . . + R ( a 2 ) + R ( a 1 ) . s=a_1+a_2+...+a_k+a_{k+1}+R(a_k)+...+R(a_2)+R(a_1). s=a1​+a2​+...+ak​+ak+1​+R(ak​)+...+R(a2​)+R(a1​).

R(S)表示将字符串逆转，如R(abcd)=dcba

题解：k+1个非空子串，非空意味着子串长度至少为1,那么如果要满足题目条件，至少前k个字符与后k个字符一一对应，并且中间 至少留下1个字符作为 a k + 1 a_{k+1} ak+1​

参考代码：

#include <iostream>
#include<bits/stdc++.h>
using namespace std;

char c[200];

void solve()
{
    int n, k;
    scanf("%d%d", &n, &k);
    getchar();
    scanf("%s", c+1);
    int cnt = 0;
    for(int i = 1; i <= n / 2; i++){//两边对称的字母个数>=k就可以
        if(c[i] == c[n-i+1])
            cnt++;
        //需要连续对称
        else
            break;
    }
    if(cnt >= k && k * 2 + 1 <= n){
        printf("YES\n");
    }
    else{
        printf("NO\n");
    }
}

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
        solve();
    return 0;
}

B. Max and Mex

题意：给你一个n个元素互不相同的集合S，会对集合S操作 k次。操作内容如下：

取集合S中没有的最小自然数a与集合S中最大的数b,往集合里 ⌈ a + b 2 ⌉ \lceil \frac{a+b}{2} \rceil ⌈2a+b​⌉

问操作k次后的集合共有多少个不同的元素。

题解：当a<b时，假设结果为c,则a<c≤b，因此，不管操作几次，集合中最小自然数和集合中的最大值都不变，即只需要判断c存不存在原集合中即可

当a=b时， ⌈ a + b 2 ⌉ \lceil \frac{a+b}{2} \rceil ⌈2a+b​⌉结果为b,元素不增加

当a>b时，即前面的元素是一个排序，a=b+1，那么 ⌈ a + b 2 ⌉ \lceil \frac{a+b}{2}\rceil ⌈2a+b​⌉会多一个元素，此时a与b都比原来的a,b多1。因此操作k次时，会多k个元素

参考代码：

#include <iostream>
#include<bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;
int a[N];

void solve()
{
    int n, k;
    scanf("%d%d", &n, &k);
    for(int i = 1; i <= n; i++)
        scanf("%d", &a[i]);

    if(k == 0){
        printf("%d\n", n);
        return;
    }

    int maxn = *max_element(a + 1, a + n + 1);
    sort(a + 1, a + n + 1, less<int>());
    int mex = n + 1;
    for(int i = 1; i <= n; i++){
        if(i-1 != a[i]){
            mex = i - 1;
            break;
        }
    }

    int mid;
    if(maxn >= mex){
        mid = (maxn + mex + 1) / 2;
        int flag = 1;
        for(int i = 1; i <= n; i++){
            if(a[i] == mid){
                flag = 0;//原集合中有这个元素
                break;
            }
        }
        printf("%d\n", n + flag);
    }
    else{
        printf("%d\n", n + k);
    }


}

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
        solve();
    return 0;
}

C. Diamond Miner

题意：有n个钻石矿和n个挖矿机，每个挖矿机只能挖一个钻石矿，假设挖矿机的位置为(a,b),钻石矿的位置是(c,d)则挖矿机挖这块矿的代价为 （ a − b ） 2 + ( c − d ) 2 \sqrt{（a-b）^2+(c-d)^2} （a−b）2+(c−d)2 ​

问挖完这n座矿的最小代价是多少。题目保证钻石矿都在x轴上，挖矿机在y轴上。

题解:分别对钻石矿和挖矿机进行排升序，小的对小的，大的对大的即可。理由如下图 。

把黄色与彩虹色维成的图形以两条彩虹边相交的点分割成两个三角形，显然两条彩虹边的长度大于两条黄边的长度，因此当大的跟大的匹配，小的跟小的匹配时，代价最小。

参考代码：

#include <iostream>
#include<bits/stdc++.h>
using namespace std;

const int N = 1e5 + 10;
typedef long long ll;
ll a[N], b[N];

void solve()
{
    int n;
    scanf("%d", &n);
    int cntx, cnty;
    cntx = cnty = 1;
    for(int i = 1; i <= 2*n; i++){
        int x, y;
        scanf("%d%d", &x, &y);
        if(x == 0){
            a[cnty++] = abs(y);
        }
        else{
            b[cntx++] = abs(x);
        }
    }

    sort(a + 1, a + n + 1);
    sort(b + 1, b + n + 1);

    double ans = 0.00;
    for(int i = 1; i <= n; i++)
    {
        ans += sqrt(a[i]*a[i] + b[i] * b[i]);
    }
    printf("%.10f\n", ans);
}

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
        solve();
    return 0;
}
