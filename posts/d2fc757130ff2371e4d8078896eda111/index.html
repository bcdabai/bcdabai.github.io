C++语言基础——C++一本通：第四章 循环结构的程序设计：第一节 for语句:1067：整数的个数


#include<bits/stdc++.h>
using namespace std;
int main()
{
  int n,sh;
    int h_1=0,h_5=0,h_10=0;
    int i;
    cin>>n;
    for(i=1; i<=n; i++)
    {
        cin>>sh;
        if(sh==1)	
            h_1++;
        if(sh==5)	
            h_5++;
        if(sh==10)	
            h_10++;
    }
    cout<<h_1<<endl;
    cout<<h_5<<endl;
	cout<<h_10<<endl;
	return 0;
}
