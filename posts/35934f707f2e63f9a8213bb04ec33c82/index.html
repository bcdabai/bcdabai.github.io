另一种方法  这里原理不在赘述

#include<stdio.h>
int main(void)
{
    int n;
	int i,j;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
		for(j=0;j<=n-i;j++)
		{
			printf(" ");
		}
        for(j=1;j<=i;j++)
        {
            printf("* ");
        }
		printf("\n");
    }
	for(i=1;i<=n+1;i++)
	printf("* ");
	printf("\n");
	for(i=n;i>=1;i--)
    {
		for(j=n-i;j>=0;j--)
		{
			printf(" ");
		}
        for(j=i;j>=1;j--)
        {
            printf("* ");
        }
		printf("\n");
    }
    return 0;
}
