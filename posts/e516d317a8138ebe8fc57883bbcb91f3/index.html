int main(void)
{
	int a[] = {11,1, 6, 3, 66, 58, 79, 33};
	int n = sizeof(a) / sizeof(a[0]);
	int j,buf;
    for (j=0; j<n; ++j){
    	if (a[j] > a[j+1]){
            buf = a[j];
            a[j] = a[j+1];
            a[j+1] = buf;
        }
    }
    
    for(j=0; j<n; j++){
		printf("%d ",a[j]);
	}
}

int main(void)
{

	int j,buf,i;
	for (i=0; i<n; i++){
		for (j=0; j<n-i; j++){
	    	if (a[j] > a[j+1]){
	            buf = a[j];
	            a[j] = a[j+1];
	            a[j+1] = buf;
	        }
	    }
	}
    
    for(j=0; j<n; j++){
		printf("%d ",a[j]);
	}
}
