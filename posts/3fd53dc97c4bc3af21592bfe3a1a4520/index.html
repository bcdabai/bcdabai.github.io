用C语言实现输入四个整数求其最大值

#include<stdio.h>
int main()
{
	printf("请输入四个数:\n");
	double a,b,c,d,max;
	scanf("%lf%lf%lf%lf",&a,&b,&c,&d);
	max=a;
	if(b>max)
		max=b;
	if(c>max)
		max=c;
	if(d>max)
		max=d;
	printf("最大值为%.2lf\n",max);
	return 0;

}
