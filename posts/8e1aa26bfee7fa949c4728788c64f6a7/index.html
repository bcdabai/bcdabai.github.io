            float: left;
            margin: 15px;

            color: #ffd026;
            font-size: 20px;
        }
        .rg_left > p:last-child{
            color: #a6a6a6;

        }

            float: right;
            margin: 15px;
        }

            font-size: 15px;
        }

            color: pink;
        }

            width: 100px;

            text-align: right;

            width: 110px;
        }
        #img_check{
            height: 32px;
            vertical-align: middle;
        }
        #btn_sub{
            width: 150px;
            height: 40px;
            background-color: #ffd026;
