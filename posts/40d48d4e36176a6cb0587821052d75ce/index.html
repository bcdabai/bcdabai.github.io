【java】窗口设计

package 我的Java;

import	javax.swing.*;
import  java.awt.*;

public class A  {
	public static void main(String args[]) {
		JFrame j=new JFrame("第一窗口");//创建窗口
		j.setSize(400,200);
		JButton jbn1=new JButton("按钮1");//创建按钮
		JButton jbn2=new JButton("按钮2");
		JButton jbn3=new JButton("按钮3");
		JButton jbn4=new JButton("按钮4");
		JButton jbn5=new JButton("按钮5");
		j.add(jbn1,BorderLayout.NORTH);//将按钮放进去
		j.add(jbn2,BorderLayout.WEST);
		j.add(jbn3,BorderLayout.CENTER );
		j.add(jbn4,BorderLayout.EAST);
		j.add(jbn5,BorderLayout.SOUTH);
		j.setVisible(true);
		}
}
