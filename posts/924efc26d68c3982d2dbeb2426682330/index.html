定义一个函数求n(阶乘）,再在主函数中通过调用该函数计算: 1（阶乘）+3（阶乘）+5（阶乘）+...+(2n-1)(阶乘）输入提示信息格式: “Input n:\n“输入数据格式要求

#include<stdio.h>
long Fact(int j);
int main()
{
    int n,j,sum=0;
    printf("Input n:\n");
    scanf("%d",&n);
    for((j=2*n-1);j>=1;j-=2)
    {
        sum+=Fact(j);
    }
    printf("sum=%ld\n",sum);
}
long Fact(int j)
{
    int i,result=1;
    for(i=1;i<=j;i++)
    {
        result*=i;
    }
    return result;

}
