import csv
def reverse_lookup(d,v):
    for k in d:
        if d[k]!=v:
            return k
with open('D:\data1.csv','r') as f:
    readers=csv.reader(f)
    d=dict()
    for row in readers:
        #print(row[0])
        if row[0] not in d:
            d[row[0]]=1
        else:
            d[row[0]]=d[row[0]]+1
    print(reverse_lookup(d, 1))
