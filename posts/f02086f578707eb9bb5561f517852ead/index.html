打印菱形

#include<stdio.h>
#include<math.h>
int main()
{
	int n;//n 为奇数，以 7 为例 
	int i = 1, j, m, N;
	scanf("%d", &N);
	
		for(j = N-1;j > 1;j -= 2){ //上面三行，空格以 2 递减 
			for(m = 0;m < j;m++){//打印第一行的 6 个空格 
				printf(" ");
			}
			for(n = 0;n < N-j;n++){//打印第一行的 1 个 * 
				printf("* ");
			}
			printf("\n");//换行，一共有三行，且每行空格数和 * 数互补 
		}
		
	for(i = 0;i < N;i++){//打印 N 个 * ，中间那行 
		printf("* ");
	}
	printf("\n");
	
		for(j = 2;j < N;j += 2){//下面三行，空格以 2 递增 
			for(m = 0;m < j;m++){//打印第一行的 2 个空格 
				printf(" ");
			}
			for(n = 0;n < N-j;n++){//打印第一行的 5 个 *  
				printf("* ");
			}
			printf("\n"); 
		}
	return 0;
 }

 思路就是先打印空格再打印“*   ”。

并且空格和“*   ”的数目相加等于N
