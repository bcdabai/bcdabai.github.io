.scroll{
    width: 100%;
    background: #FFEAAE;
    height: 35px;
    line-height: 35px;
    color: #9C5629;
    padding: 0 30px 0 20px;
    position: relative;
  }
  .close{
    display: block;
    position: absolute;

    text-align: center;
    right: 0;
    bottom: 0;
  }

    overflow: hidden;

  }

    animation-play-state:paused;
  }
