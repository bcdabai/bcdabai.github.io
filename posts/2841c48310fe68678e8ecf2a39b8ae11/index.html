【C语言练习题】abc+cba=1333

 已知a，b，c都是1位整数，求当三位整数abc、cba的和为1333时a、b、c的值。

#include<stdio.h>
int main()
{
    int i,j,k;
    for(i=0;i<10;i++)
    {
        for(j=0;j<10;j++)
        {
            for(k=0;k<10;k++)
            {
                if(i*100+j*10+k+k*100+j*10+i==1333)
                {
                    printf("%d %d %d",i,j,k);
                    printf("\n");
                    printf("-----\n");
                }
            }
        }
    }
    return 0;
}
