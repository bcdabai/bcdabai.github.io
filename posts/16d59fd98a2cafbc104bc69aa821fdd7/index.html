最简单的找质因子的方法

诺。。。直接小代码一贴。。

int main()
{
    int n;
    while(cin>>n)
    {
        for(int d=2;d*d<=n;d++)
        {
            while(n%d==0)
            {
                cout<<d<<" ";
                n/=d;
            }
        }
        if(n!=1)
            cout<<n;
        cout<<endl;
    }
    return 0;
}
