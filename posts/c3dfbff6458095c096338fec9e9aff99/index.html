给出 n，请输出一个直角边长度是 n 的数字直角三角形。所有数字都是 2 位组成的，如果没有 2 位则加上前导 0

#include<stdio.h>
int main()
{   int n,cnt=0;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        for(int j=i;j<=n;j++) //输入左上直角三角形的方法 
        {
            cnt++; //计数都要初始化 计数也可以用来输入按顺序的数字 新思路get！ 
            if(cnt<10)//判断是不是两位数 
            {
                printf("%02d",cnt); //两位 没有两位就用0补齐 
            }
            else
            {
                printf("%d",cnt);
            }                        
        }
        printf("\n");//打印三角形都要在第一层大循环之后换行 
    }
    return 0;
}
