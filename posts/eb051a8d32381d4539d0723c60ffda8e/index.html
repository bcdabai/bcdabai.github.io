C语言 完全平方数

若一个数能表示成某个数的平方的形式，则称为完全平方数。

#include<stdio.h>
#include<math.h>
int IsSquare(long int n);
int main() {
	long n,p;
	scanf("%d",&n);
	if(p==IsSquare(n))
		printf("%d 不是平方数",n);
	else
		printf("%d 是平方数\n",n);
	return 0;
}

int IsSquare(long int n) {
	int m,t;
	m=(int)sqrt(n);
	t=(int)sqrt(n)+1;
	if(m*m==n)
		return m;
	else if(t*t==n)
		return t;
	else
		return 0;
}
