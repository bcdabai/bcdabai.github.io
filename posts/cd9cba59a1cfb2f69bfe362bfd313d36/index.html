查询当前数据库所有表的数据量

USE 当前数据库名称
GO

SELECT A.NAME ,B.ROWS  
FROM sysobjects  A JOIN sysindexes B ON A.id = B.id 
WHERE A.xtype = 'U' AND B.indid IN(0,1) 
ORDER BY B.ROWS DESC，A.NAME
