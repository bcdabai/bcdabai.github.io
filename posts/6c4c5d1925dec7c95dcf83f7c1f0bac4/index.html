运用for循环找到n个数中的最大数

#include<stdio.h>
int main()
{
    int i,n;
    float x,max;
    printf("请输入第1个数:");
    scanf("%f",&x);
    max=x;

printf("请输入n的值”);

scanf("%d",n);
    for(i=1;i<n;i++)
    {
        printf("请输入第%d个数：",i+1);
        scanf("%f",&x);
        if(x>max)
           max=x;
    }
    printf("10个数的最大值是；%.0f",max);
}
