# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def getIntersectionNode(self, headA: ListNode, headB: ListNode) -> Optional[ListNode]:
        if headA is None or headB is None :
            return None
        tmpA,tmpB = headA,headB
        while tmpA!=tmpB :
            if tmpA is None:
                tmpA=headB
            else :
                tmpA=tmpA.next
            if tmpB is None:
                tmpB=headA
            else :
                tmpB=tmpB.next
        return tmpB
