Java中矩阵运算(math3的使用)

math3简介

math3是Apache下的一款进行数学计算的一款java开源工具。jar包名称为：commons-math3-3.6.1.jar。大家可以通过maven或者其他网站进行下载。

math3是一款非常好用的工具，里面提供了各种运算的方法及类，方便大家调用。其API文档为：http://commons.apache.org/proper/commons-math/javadocs/api-3.2/index.html。可以参照API文档进行学习，也可以在实战中，进行学习。比如，从正太分布中采样，从逆威沙特分布分布中采样等等，都可以基于此工具完成。具体可参考这一篇博客http://blog.csdn.net/qy20115549/article/details/54317981。

math3中的矩阵操作

由于在写算法时，经常遇到矩阵的运算，如数组转化为矩阵，矩阵的逆，矩阵的转置，矩阵的乘法，矩阵转为数组等等。下面写了一个小程序演示一下，相关操作。具体请看注释。

 
 
  1. package sampling.method;
  3. import org.apache.commons.math3.linear.Array2DRowRealMatrix;
  4. import org.apache.commons.math3.linear.LUDecomposition;
  5. import org.apache.commons.math3.linear.RealMatrix;
  6. /*
  7.  * author:合肥工业大学 管院学院 钱洋 
  8.  *1563178220@qq.com
  9.  *博客地址:http://blog.csdn.net/qy20115549/
  10. */
  11. public class MathMethod {
  12.     public static void main(String args[]){
  13.         double b [][] = new double[5][5];
  14.         for(int i = 0; i < b.length; i++) {
  15.             b[i][i] = 1;  
  16.         }
  17.         //将数组转化为矩阵
  18.         RealMatrix matrix = new Array2DRowRealMatrix(b);
  19.         System.out.println("创建的数组为：\t"+matrix);
  20.         //获取矩阵的列数 getColumnDimension() 
  21.         System.out.println("矩阵的列数为:\t"+matrix.getColumnDimension());
  22.         //获取矩阵的行数
  23.         System.out.println("矩阵的行数为:\t"+matrix.getRowDimension());
  24.         //获取矩阵的某一行,返回,仍然为矩阵
  25.         System.out.println("矩阵的第一行为:\t"+ matrix.getRowMatrix(0));
  26.         //获取矩阵的某一行,返回,转化为向量
  27.         System.out.println("矩阵的第一行向量表示为:\t"+ matrix.getRowVector(1) );
  28.         //矩阵的乘法
  29.         double testmatrix[][] = new double[2][2];
  30.         testmatrix[0][0] = 1;
  31.         testmatrix[0][1] = 2;
  32.         testmatrix[1][0] = 3;
  33.         testmatrix[1][1] = 4;
  34.         RealMatrix testmatrix1 = new Array2DRowRealMatrix(testmatrix);
  35.         System.out.println("两个矩阵相乘后的结果为：\t"+testmatrix1.multiply(testmatrix1) );
  36.         //矩阵的转置
  37.         System.out.println("转置后的矩阵为：\t"+testmatrix1.transpose());
  38.         //矩阵求逆
  39.         RealMatrix inversetestMatrix = inverseMatrix(testmatrix1);
  40.         System.out.println("逆矩阵为：\t"+inversetestMatrix);
  41.         //矩阵转化为数组 getdata
  42.         double matrixtoarray[][]=inversetestMatrix.getData();
  43.         System.out.println("数组中的某一个数字为：\t"+matrixtoarray[0][1]);
  44.     }
  45.     //求逆函数
  46.     public static RealMatrix inverseMatrix(RealMatrix A) {
  47.         RealMatrix result = new LUDecomposition(A).getSolver().getInverse();
  48.         return result; 
  49.     }
  50. }
