C语言，输入4个整数，输出最大值

1，简单型 

#include<stdio.h>
int main()
{
    int a, b, c, d,i;        //这里的i作为一个交换数值作用
	scanf("%d %d %d %d", &a, &b, &c, &d);
	if (a < b)
	{
		i = b;
		b = a;
		a = i;
	}
	if (a < c)
	{
		i = c;
		c = a;
		a = i;
	}
	if (a < d)
	{
		i = d;
		d = a;
		a = i;
	}
	printf("%d\n", a);
    return 0;
}

2.较简单型

#include<stdio.h>
int main()
{
    int arr[4] = { 0 };
	int i = 0;
	while (i < 4)
	{
		scanf("%d", &arr[i]);
		i++;
	}
	int max = arr[0];	//假设数组中第一个为最大值
	int j = 1;
	while (j< 4)
	{
		if (arr[0] < arr[j])
		{
			max = arr[j];
		}
		j++;
	}
	printf("%d\n", max);
    return 0;
}
