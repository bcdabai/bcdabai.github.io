kindle剪贴内容提取

file_path = 'C:/Users/74285/Desktop/'
file = open(file_path + 'My Clippings.txt', 'r', encoding = 'utf-8')
data = {}
while 1:
    line = file.readline()
    if not line:
        break
    book = line
    if book not in data:
        data[book] = []
    line = file.readline()
    line = file.readline()
    line = file.readline()
    data[book].append(line) 
    line = file.readline()
file.close()
books = data.keys()
for book in books:
    path = file_path + book[: -1] + '.txt'
    book_file = open(path, 'w', encoding = 'utf-8')
    for line in data[book]:
        book_file.write(line)
    book_file.close()
