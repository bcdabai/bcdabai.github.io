关于矩阵的编程题（C语言版，持续更新~~~~~~）

题目汇总：

  1. 矩阵对角线求和（分别求出主对角线和副对角线元素的和）
  2. 矩阵对角线求和（主对角线和副对角线元素总和）
  3. 矩阵转置（n阶）
  4. 矩阵转置（n*m阶）
  5. 输入一个n*m的矩阵，求出每行的最大值和每行的总和。要求把每行总和放入每行最大值的位置，如果有多个最大值，取下标值最小的那一个作为最大值。
  6. 输入一个正整数n (1≤ n ≤6),再输入一个n 行n列的矩阵，找出该矩阵中绝对值最大的元素以及它的行下标和列下标。

第1题：

解法思路：
  这里通过使用二维数组分别求出矩阵主对角线和副对角线元素的和。本题以三阶矩阵为例。在线性代数中，矩阵的下标是从1开始的，例如a[1][1]表示第一行第一列的元素。但是如果在使用二维数组的情况下，矩阵的下标是从0开始的，a[0][0]表示第一行第一列的元素。
  主对角线上元素的下标我们可以直观的发现的行列坐标是相等的，在二维数组和线性代数的情况下都表现为 i==j
  副对角线上的下标之和用公式表现为 i+j == n - 1，其中n表示矩阵的阶数（二维数组的情况下）
  副对角线上的下标之和用公式表现为 i+j == n + 1，其中n表示矩阵的阶数（线性代数的情况下）

代码如下：

#include<stdio.h>
int main()
{
    int a[3][3];	//定义一个3*3的数组
    int i,j,sum1 = 0,sum2 = 0;//sum1保存主对角线元素和，sum2保存副对角线元素和
    for(i=0; i<3; i++){		//通过两个for循环来实现n阶矩阵的键盘输入
        for(j=0; j<3; j++){
            scanf("%d",&a[i][j]);
        }
    }
    for(i=0; i<3; i++){		//遍历数组，筛选出满足条件的元素
        for(j=0; j<3; j++){
            if(i == j)
                sum1 += a[i][j];	//主对角线元素和
            if(i + j == 2)
                sum2 += a[i][j];	//副对角线元素和
        }
    }
    printf("%d %d\n",sum1,sum2);
    return 0;
}


第2题：

解法1：
  第2题与第1题很类似，这里通过使用二维数组求出矩阵主对角线和副对角线的元素总和。
  主对角线上元素的下标我们可以直观的发现它的行列坐标是相等的，在二维数组和线性代数的情况下都表现为 i==j
  副对角线上的下标之和用公式表现为 i+j == n - 1，其中n表示矩阵的阶数（二维数组的情况下）
  副对角线上的下标之和用公式表现为 i+j == n + 1，其中n表示矩阵的阶数（线性代数的情况下）

解法2（不使用数组）：
（1）设元素下标为i,j ，矩阵阶数为Order，主对角线和副对角线上的元素总和为sum；
（2）主对角线元素下标满足：i == j；
（3）副对角线元素下标满足：i+j == Order + 1；（注意：在没有使用数组的情况下，我们采用线性代数中的矩阵下标，从1开始）
（4）输入一个矩阵元素 Element，判断下标是否满足主副对角线元素下标的条件，满足，sum加上这个数；

解法1的代码如下：

#include <stdio.h>
int main()
{
    int i,j;
    int a[10][10];	//定义一个二维数组
    int sum = 0;	//sum表示主对角线和副对角线元素总和
    int n = 0;		//n表示键盘输入的矩阵阶数
    printf("请输入矩阵的阶数：");
    scanf("%d",&n);
    for(i=0; i<n; i++){	//通过两个for循环来实现n阶矩阵的键盘输入
        for(j=0; j<n; j++){
            scanf("%d",&a[i][j]);
        }
    }
//满足主对角线条件：i==j; 满足副对角线条件：i+j == n-1;
    for(i=0; i<n; i++){
        for(j=0; j<n; j++){
            if( (i==j) || (i+j == n-1) ){
                sum += a[i][j];
            }
        }
    }
    printf("%d\n",sum);
    return 0;
}

解法2的代码如下：

#include <stdio.h>
int main(){
    int sum = 0, Order, Element;//Order表示矩阵的阶数，Elenent表示矩阵元素
    printf("请输入矩阵的阶数："); 
    scanf("%d", &Order);                     // 阶数
    printf("请输入矩阵的各个元素，输入的元素个数为矩阵阶数的平方："); 
    for(int i = 1; i<=Order; i++){        // i从1开始，注意是<=
        for(int j = 1; j<=Order; j++){    // j从1开始，注意是<=
            scanf("%d",&Element);
            if ( (i == j) || (i + j) == (Order + 1) ) //判断
                sum += Element;                       //求和
        }
    }
    printf("主副对角线上的元素总和为：%d\n",sum);
    return 0;
}


第3题：

解法思路：
   以三阶矩阵为例：
   1 2 3        1 4 7
   4 5 6 转置后  2 5 8
   7 8 9        3 6 9
不难发现，转置后的矩阵，主对角线上的元素没有发生变化，所以可以进行判断，例如 a[i][j] ,当某个元素的横坐标小于纵坐标时，也就是 i < j 时，将a[i][j] 与a[j][i] 进行交换，交换时需要用到temp这个中间变量。

代码如下：

#include <stdio.h>
int main()
{
    int a[100][100];	// 声明一个数组
	int temp, i, j, n;	// n代表矩阵的阶数,temp是中间变量
	printf("输入矩阵的阶数：");
    scanf("%d", &n);
	printf("请输入一个%d阶矩阵：\n",n);
    for(i=0; i<n; i++){		// 通过两个for循环进行数组的键盘输入
        for(j=0; j<n; j++){
            scanf("%d", &a[i][j]);
        }
    }
	printf("转置后的矩阵为：\n");
    for(i=0; i<n; i++){
        for(j=0; j<n; j++){
            if(i<j){	// 因为n阶矩阵转置,主对角线上的元素是不变的，所以是i<j
                temp = a[i][j]; 
				a[i][j] = a[j][i]; 
				a[j][i] = temp;
            }
            printf("%d ", a[i][j]);
        }
        printf("\n"); // 内层for循环结束后，进行换行，避免矩阵在同一行输出
    }
    return 0;
}


第4题：

解法思路：
  与第3题很类似，解法有多种

代码如下：

#include <stdio.h>
int main(){
	int n, m,i,j;	// n代表行数，m代表列数
	printf("输入矩阵的行数n和列数m：");
	scanf("%d%d", &n, &m);
	int arr[20][20];
	printf("请输入一个%d*%d阶的矩阵：\n",n,m);
	for(i=0; i<n; i++){
		for(j=0; j<m; j++){
			scanf("%d", &arr[i][j]);
		}
	}
	printf("转置后的矩阵为：\n");	// 转置后的矩阵是m*n阶
	for(i=0; i<m; i++){			// 注意外层循环是i<m
		for(j=0; j<n; j++){		// 注意内层循环是j<n
			printf("%d ", arr[j][i]); 
			// 因为数组下标是从0开始,这里判断j==n-1,是为了避免转置后的矩阵在同一行输出
			if(j==(n-1)){		
				printf("\n");
			}
		}
	}
	return 0;
}


第5题：

解法思路：
  在代码中进行了详细解释

代码如下：

#include<stdio.h>
/*
编写一个程序输入一个n*m的矩阵存储并输出，并且求出每行的最大值和每行的总和。
要求把每行总和放入每行最大值的位置，如果有多个最大值，取下标值最小的那一个作为最大值。
最后将结果矩阵输出。
*/
int main(){
    int n,m,i,j,max,t,sum;	// n行数，m列数，t用来保存矩阵每行元素中最大元素所在的列数,max表示每行元素的最大值,sum表示矩阵每行元素的总和
	int a[100][100];
	printf("请输入矩阵的行数n和列数m：");	
	scanf("%d%d",&n,&m);
	for(i=0; i<n; i++){
		for(j=0;j<m;j++){
			scanf("%d",&a[i][j]);
		}
	}
    for(i=0; i<n; i++){
		max = a[i][0];	// 第一次循环时i的值0，先假设第一行中第一个元素是最大的(数组下标从0开始)
		t = 0;		    // 内层for循环完成后，需要将t置为0,开始第二轮循环(也就是计算矩阵第二行元素中最大元素所在的列数)
        sum = 0;        // 内层for循环完成后，需要将sum置为0,开始第二轮循环(也就是计算矩阵第二行元素的总和)
        for(j=0; j<m; j++){
			if(a[i][j]>max){
				max = a[i][j];
                t = j;
			}
			sum += a[i][j];
		}
		a[i][t] = sum;	// 将矩阵每行元素的总和保存在矩阵每行最大元素所在的位置上
	}
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			printf("%d ",a[i][j]);
		}
		printf("\n");	// 避免矩阵在同一行输出
    }
    return 0;
}


第6题：

解法思路：
  注意题目中的绝对值，对于整形的（即int型），取绝对值用 abs()函数；对于浮点型的即（double 或者 float），取绝对值用 fabs()函数；
  其它的注意事项在代码中的注释中有说明；

代码如下：

/*
输入一个正整数n (1≤ n ≤6),再输入一个n行n列的矩阵，
找出该矩阵中绝对值最大的元素以及它的行下标和列下标。
*/
#include<stdio.h>
#include<math.h>
int main()
{
	int n,i,j,max=0,row=0,column=0;
	int a[6][6];
	printf("请输入矩阵的阶数：\n");
	scanf("%d",&n);
	printf("请输入一个%d*%d的矩阵：\n",n,n);
	for(i=0;i<n;i++)	// 输入一个n阶矩阵
	{
		for(j=0;j<n;j++)
		{
			scanf("%d",&a[i][j]);
		}
	}
	for(i=0; i<n;i++)
	{
		for(j=0; j<n; j++)
		{
			if(abs(a[i][j]) > max)
			{
				max = abs(a[i][j]); // 这里是将abs(a[i][j])的值赋给max，不是将a[i][j]赋给max
				row = i;
				column = j;
			}
		}
	}
	//输出绝对值最大的值,这里用的是a[row][cloumn],而不是max,因为max的值是取绝对值之后的值，并不是原先的值。
	//因为数组下标是从0开始的,所以输出行数和列数时row和column要加1 
    printf("绝对值最大的元素是 %d\n所在的行和列分别为 %d  %d\n",a[row][column],row+1,column+1); 
    return 0;
}
