使用scanf向c语言的结构体数组输入数据

# include <stdio.h>
struct STU
{
    char name[10];
    int age;
    char sex;
    char num[20];
};
int main()
{
    struct STU stu[3];
    scanf ("%s%d %c%s", stu[0].name, &stu[0].age, &stu[0].sex, stu[0].num);/*%c前面要加空格, 不然输入时会将空格赋给%c*/
   	printf("学生姓名：%s 学生年龄：%d 学生性别：%c 学生学号：%s\n", stu[0].name, stu[0].age, stu[0].sex, stu[0].num);
}
