C语言，输入身高（cm），体重（kg），计算身体的BMI值


#include<stdio.h>
int main()


{
    int h = 0;
    int w = 0;
    double BMI = 0.0;
    scanf("%d %d", &h, &w);
    BMI = w/(h/100.0)/(h / 100.0);
    printf("%.3lf\n", BMI);
	return 0;
}
