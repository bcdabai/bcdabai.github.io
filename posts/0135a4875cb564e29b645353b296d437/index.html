SQL Server | 设置默认值

1、建表时设置默认值

Create table Sys_User (
   Id int primary key identity,
   Name  nvarchar(50),
   Gender char(2) default '男',
   CreationTime datetime default getdate()
)

-- 添加一条测试数据
