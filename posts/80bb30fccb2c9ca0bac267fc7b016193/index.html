c语言判断一个数是偶数还是奇数？

#include<stdio.h>

int main()

{
    int a = 0;        //定义一个变量a，并赋值为0；
    scanf("%d", &a);        //输入a的值；
    if (a % 2 == 0)        //判断a除以2是否有余数，0为没有余数，满足条件时，输出下面的值；
        printf("%d是偶数",a);        //满足if条件时输出；
    else        //不满足if条件时，输出下面
        printf("%d是奇数",a);        //输出
    return 0;

}
