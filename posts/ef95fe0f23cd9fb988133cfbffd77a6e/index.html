洛谷入门B2005《打印数字三角形》

# include <stdio.h>
void tra(int n,char ch){
	int i,j;
	for(i=1;i<=n;i++){
		for(j=1;j<=n-i;j++)  #每行打印n-i个空格
			printf(" ");
		for(j=1;j<=2*i-1;j++) #每行打印2*i-1个字符
			printf("%c",ch);
		printf("\n");
	}
}
int main (void){
	int n;
	char ch;
	scanf("%d %c",&n,&ch);
	tra(n,ch);
    return 0;
}
