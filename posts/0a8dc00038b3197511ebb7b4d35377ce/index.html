题目要求

1、打印金字塔
2、键盘输入金字塔的层数

一、实现代码

#include<stdio.h>
int main()
{
	int j = 1, i = 1, count = 3;//1、1默认层数为3
	int k = 0,empty = 0; //2.1、初始化空格 
	scanf("%d",&count);//3、键盘输入 
	for (j = 1; j <= count; j++)//金字塔层数
	{
		 for (empty = 0; empty < count - j; empty++)//2.2打印空格 
		 {
		 	printf(" ");
		 } 
		
		k = 2 * j - 1;//1.3因为for循环的表达式1会把i初始化所以用k代表结束条件 
		for (i = 1; i <= k; i++)//1.2每层金字塔的个数 
		{
			printf("*");
		}
		printf("\n");//换行	
	} 
}
