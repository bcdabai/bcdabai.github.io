Java基于UDP协议点对点聊天

Java基于UDP协议实现点对点聊天，只要有一方输入88，双方就都下线了，下面是代码：
发送端代码(简称甲方)：

package Demo0915;

import java.net.DatagramPacket;
import java.net.DatagramSocket;
import java.net.InetAddress;
import java.util.Scanner;

/**
 * <p>基于UDP协议发送消息者</p>
 *
 * @author 波波老师(weixin : javabobo0513)
 */
public class SenderUDP {
    public static void main(String[] args) throws Exception {
        DatagramSocket ds = new DatagramSocket();
        //创建目的地对象
        InetAddress ia = InetAddress.getByName("192.168.2.95");
        Scanner sc = new Scanner(System.in);
        byte[] contentArray = new byte[1024];
        //创建数据包
        DatagramPacket dp = new DatagramPacket(contentArray, contentArray.length, ia, 48080);

        byte[] content = new byte[1024];
        //2、创建接收数据的数据包
        DatagramPacket dp2 = new DatagramPacket(content, content.length);
        DatagramSocket ds2 = new DatagramSocket(48081);
        while (true){
            System.out.println("请输入要对乙方说的话：");
            //接收用户输入的消息
            String str = sc.next();
            dp.setData(str.getBytes(), 0, str.getBytes().length);
            //甲方发送消息
            ds.send(dp);
            if("88".equals(str)){
                break;
            }
            System.out.println("消息发送完毕，等待乙方的回答：");
            //接收乙方发送的消息
            ds2.receive(dp2);
            InetAddress address = dp2.getAddress();
            if(address != null){
                if(!"192.168.2.95".equals(address.getHostAddress())){
                    System.out.println("搞事情的人IP=" + address.getHostAddress());
                    continue;
                }
            }
            byte[] data = dp2.getData();
            String s = new String(data, 0, dp2.getLength());
            System.out.println("乙方答复消息说：" + s);
            if("88".equals(s)){
                break;
            }
        }
        System.out.println("下次再见");
        //关闭资源
        ds.close();
    }
}

接收端代码(简称乙方)：

package Demo0915;

import java.io.IOException;
import java.net.DatagramPacket;
import java.net.DatagramSocket;
import java.net.InetAddress;
import java.util.Scanner;

/**
 * <p>基于UDP协议接收消息端</p>
 *
 * @author 波波老师(weixin : javabobo0513)
 */
public class ReceiveUDP {

    public static void main(String[] args) throws IOException {
        //1、创建Socket对象，指定端口（和发送端保持一致）
        DatagramSocket ds = new DatagramSocket(48080);
        byte[] content = new byte[1024];
        //2、创建接收数据的数据包
        DatagramPacket dp = new DatagramPacket(content, content.length);

        DatagramSocket ds2 = new DatagramSocket();
        //创建目的地对象
        InetAddress ia = InetAddress.getByName("192.168.2.95");
        Scanner sc = new Scanner(System.in);
        byte[] contentArray = new byte[1024];
        //创建数据包
        DatagramPacket dp2 = new DatagramPacket(contentArray, contentArray.length, ia, 48081);

        while (true){
            System.out.println("我是乙方，准备接收消息....");
            //3、socket对象调用receive方法，接收数据
            ds.receive(dp);
            InetAddress address = dp.getAddress();
            if(address != null){
                if(!"192.168.2.95".equals(address.getHostAddress())){
                    System.out.println("搞事情的人IP=" + address.getHostAddress());
                    continue;
                }
            }
            //4、获取接收到的数据
            byte[] data = dp.getData();
            //接收到的数据
            String s = new String(data, 0, dp.getLength());
            //5、处理数据
            System.out.println("接收到甲方数据，他说：" + s);
            if("88".equals(s)){
                break;
            }
            System.out.println("请输入要对甲方说的话：");
            //接收用户输入的消息
            String str = sc.next();
            dp2.setData(str.getBytes(), 0, str.getBytes().length);
            //甲方发送消息
            ds.send(dp2);
            if("88".equals(str)){
                break;
            }
            System.out.println("消息发送完毕，等待甲方的回答：");
        }
        System.out.println("下次再见");
        ds.close();
    }
}

注意：先运行接收端，再运行发送端
