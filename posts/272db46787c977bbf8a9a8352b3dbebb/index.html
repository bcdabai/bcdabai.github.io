一、前期准备工作

1、开发使用工具：IDEA、Mysql

二、开始流程

1、使用IDEA创建空的javase工程
2、创建完成后引入依赖

<dependencies>
        <!-- https://mvnrepository.com/artifact/org.mybatis/mybatis -->
        <dependency>
            <groupId>org.mybatis</groupId>
            <artifactId>mybatis</artifactId>
            <version>3.5.2</version>
        </dependency>

        <dependency>
            <groupId>mysql</groupId>
            <artifactId>mysql-connector-java</artifactId>
            <scope>runtime</scope>
            <version>8.0.18</version>
        </dependency>
</dependencies>

3、在pom.xml中增加下方代码（target中找不到mapper.xml使用）

<build>
        <resources>
            <resource>
                <directory>src/main/java</directory>
                <includes>
                    <include>**/*.xml</include>
                </includes>
            </resource>
        </resources>
</build>

4、工程目录

5、代码一之User.java

package org.javayu.Model;

public class User {

    private Integer id;
    private String username;
    private String address;

    public Integer getId() {
        return id;
    }

    public void setId(Integer id) {
        this.id = id;
    }

    public String getUsername() {
        return username;
    }

    public void setUsername(String username) {
        this.username = username;
    }

    public String getAddress() {
        return address;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    @Override
    public String toString() {
        return "User{" +
                "id=" + id +
                ", username='" + username + '\'' +
                ", address='" + address + '\'' +
                '}';
    }
}

6、代码二之Usermapper.xml

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd" >
<mapper namespace="org.javayu.Model.UserMapper">

    <select id="getUserById" resultType="org.javayu.Model.User">
            select *from user where id=#{id};
    </select>
</mapper>

10、实现增查改删

  • 代码UserMpper.xml
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd" >
<!--命名空间指向XML索要的接口-->
<mapper namespace="org.javayu.Model.UserMapper">

    <!--增加  parameterType传入参数类型-->
    <insert id="addUser" parameterType="org.javayu.Model.User">
        insert into user (username,address) values (#{username},#{address})
    </insert>
    <!--查-->
    <select id="getUserById" resultType="org.javayu.Model.User">
            select *from user where id=#{id};
    </select>
    <!--改-->
    <update id="updateUser" parameterType="org.javayu.Model.User">
        update user set username=#{username} where id=#{id}
    </update>
    <!--删-->
    <delete id="deleteUserById" parameterType="org.javayu.Model.User">
        delete from user where id=#{id}
    </delete>
</mapper>

11、使用mapper接口形式实现

  • 代码UserMpper.xml
修改命名空间
<mapper namespace="org.javayu.mapper.UserMapper">
  • 代码config.xml
增加下述代码：
<!--批量加载mapper接口-->
<package name="org.javayu.mapper"/>
  • 增肌代码UserMapper.java
package org.javayu.mapper;

import org.javayu.Model.User;

public interface UserMapper {

    //增
    Integer addUser(User user);
    //查
    User getUserById(Integer id);
    //改
    Integer updateUser(User user);
    //删
    Integer deleteUserById(Integer id);
}
  • 测试代码TestMybatis.java
import org.apache.ibatis.io.Resources;
import org.apache.ibatis.session.SqlSession;
import org.apache.ibatis.session.SqlSessionFactory;
import org.apache.ibatis.session.SqlSessionFactoryBuilder;
import org.javayu.Model.User;
import org.javayu.mapper.UserMapper;

import java.io.IOException;
import java.io.InputStream;

public class TestMybatis {

    public static void main(String[] args) throws IOException {

        String resourse = "config.xml";
        InputStream inputStream = Resources.getResourceAsStream(resourse);
        SqlSessionFactory sqlSessionFactory = new SqlSessionFactoryBuilder().build(inputStream);
        SqlSession session = sqlSessionFactory.openSession();

        //使用mapper接口查询
        UserMapper mapper = session.getMapper(UserMapper.class);
        User user = mapper.getUserById(2);
        System.out.println(user.getUsername());
        session.commit();
        session.close();
    }
}

注意：

  • 1、config.xml代码映射问题
<!-- 使用相对于类路径的资源引用 -->
<mappers>
  <mapper resource="org/mybatis/builder/AuthorMapper.xml"/>
  <mapper resource="org/mybatis/builder/BlogMapper.xml"/>
  <mapper resource="org/mybatis/builder/PostMapper.xml"/>
</mappers>

<!-- 使用完全限定资源定位符（URL） -->
<mappers>
  <mapper url="file:///var/mappers/AuthorMapper.xml"/>
  <mapper url="file:///var/mappers/BlogMapper.xml"/>
  <mapper url="file:///var/mappers/PostMapper.xml"/>
</mappers>

<!-- 使用映射器接口实现类的完全限定类名 -->
<mappers>
  <mapper class="org.mybatis.builder.AuthorMapper"/>
  <mapper class="org.mybatis.builder.BlogMapper"/>
  <mapper class="org.mybatis.builder.PostMapper"/>
</mappers>

<!-- 将包内的映射器接口实现全部注册为映射器 -->
<mappers>
  <package name="org.mybatis.builder"/>
</mappers>
  • 2、mapper.xml的命名空间的问题
