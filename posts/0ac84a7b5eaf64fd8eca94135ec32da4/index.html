QT的单例模式

类型1：

classA.h

#include <QObject>

class classA ： public QObject

{

    Q_OBJECT

public:

    static classA* instance(){static classA m_classA; return &m_classA;}

    ~classA();

private:

    explicit classA();//explicit classA(QObject *parent = 0);

}
