专属程序员的浪漫

#include<stdio.h>
#include<stdlib.h>
#include<Windows.h>
int main()
{
	float x,y,a;
	for(y=1.5f;y>-1.5f;y-=0.1f)
	{
		for(x=-1.5f;x<1.5f;x+=.05f)
		{
			a=x*x+y*y-1;
			putchar(a*a*a-x*x*y*y*y<0.0f?'x':' ');
		}
		Sleep(100);
		putchar('\n');
	}
	getchar();
	system("pause");
	return 0;
}
