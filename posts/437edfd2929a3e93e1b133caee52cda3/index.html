用captcha.image下的 ImageCaptcha 生成字符验证码图片,只需三行代码

#导入captcha包下的image文件中的ImageCaptcha类,使用之前先实例化

from captcha.image import ImageCaptcha

ic = ImageCaptcha()
ic.write('1a8c', '1a8c' + '.png', format='png')
