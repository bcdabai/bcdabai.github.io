方法2

**思路：分别把奇数和偶数取出来在排序

#include <stdio.h>
int main()
{
    int a[10];//存放整体个数
    int b[10];//存放偶数
    int c[10];//存放奇数
    int i,j,k,l;
    int temp;
    int n;
    while(1)
    {
    	j = 0;
		k = 0; 
    printf("请输入10个数:");
    for(i = 0;i < 10;i ++)
    {
        scanf("%d",&a[i]);
        if(a[i] % 2 == 0)
        {
            c[j] = a[i];
            j ++;
        }
        else
        {
            b[k] = a[i];
            k ++;
        }
    }
    n = j;
    //奇数排序
    for(k = 0;k < 10 - n - 1;k ++)
    {
        for(l = 0;l < 10 - n - k - 1;l ++)
        {
            if(b[l] > b[l + 1])
            {
                temp = b[l];
                b[l] = b[l + 1];
                b[l + 1] = temp;
            }
        }
    }
    //偶数排序
    for(j = 0;j < n - 1;j ++)
    {
        for(i = 0;i < n - 1 - j;i ++)
        {
            if(c[i] > c[i + 1])
            {
                temp = c[i];
                c[i] = c[i + 1];
                c[i + 1] = temp;
            }
        }
    }
    //先打印奇数，再打印偶数
    for(i = 0;i < 10 - n;i ++)
    {
        printf("%d ",b[i]);
    }
    for(i = 0;i < n;i ++)
    {
        printf("%d ",c[i]);
    }
    printf("\n%%%%####$$$$**********************\n");
    }
    return 0;

}
