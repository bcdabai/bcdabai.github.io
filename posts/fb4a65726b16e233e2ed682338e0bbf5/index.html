1.开始菜单

就是switch游戏选择，注意input取值选择

int main ()
{
	
	int input = 0;
	do
	{
		menu();
		printf("选择：");
		scanf("%d",&input);
		switch (input)
		{
			case 1:
				game();
				break;
			case 0:
				printf("游戏结束\n");
				break;
			default:
				printf("重新选择\n");
				break;
		}
	} while (input);
	
	return 0;
}

2.初始化和打印

注意-和|打印

void Init(char board[ROW][COL], int row, int col)
{
    for (int i = 0; i < ROW; i++)
    {
        for (int j = 0; j < COL; j++)
        {
            board[i][j] = ' ';
        }
    }
}

void Print(char board[ROW][COL], int row, int col)
{
    for (int i = 0; i < ROW; i++)
    {
        for (int j = 0; j < COL; j++)
        {
            printf(" %c ",board[i][j]);
            if (COL - j > 1)
            {
                printf("|");
            }
        }
        printf("\n");
        if (ROW - i > 1)
        {
            for (int k = 0; k < COL; k++)
            {
                printf("---");
                if (COL - k > 1)
                {
                    printf("|");
                }
            }
            printf("\n");
        }
    }
}

3.玩家和电脑输入

玩家输入要注意输入合法性，电脑输入要注意随机性

void Player_Input(char board[ROW][COL], int row, int col)
{
    int x = 0, y = 0;
    while(1)
    {
        printf("请选择坐标:");
        scanf("%d %d", &x, &y);
        if (x >= 1 && x <= ROW && y >= 1 && y <= COL)
        {
            if (board[x - 1][y - 1] != ' ')
            {
                printf("位置被占\n");
            }
            else
            {
                board[x - 1][y - 1] = '*';
                break;
            }
        }
        else
        {
            printf("超出范围!\n");
        }
    }

}

void Computer_Input(char board[ROW][COL], int row, int col)
{
    printf("computer:\n");
    while (1)
    {
        int x = rand() % ROW + 1;
        int y = rand() % COL + 1;
        if (board[x - 1][y - 1] == ' ')
        {
            board[x - 1][y - 1] = '#';
            break;
        }
        
    }
}

4.胜利判断

横着的竖着的和斜着的

char Is_win(char board[ROW][COL], int row, int col)
{
    for (int i = 0; i < ROW; i++)
    {
        if (board[i][0] == board[i][1] && board[i][1] == board[i][2] && board[i][2] != ' ')
            return board[i][0];
    }
    for (int j = 0; j < COL; j++)
    {
        if (board[0][j] == board[1][j] && board[1][j] == board[2][j] && board[j][2] != ' ')
            return board[0][j];
    }
    if (board[0][0] == board[1][1] && board[0][0] == board[2][2] && board[2][2] != ' ')
    {
        return board[0][0];
    }
    if (board[0][2] == board[1][1] && board[0][2] == board[2][1] && board[0][2] != ' ')
    {
        return board[0][2];
    }
    for (int k = 0; k < ROW; k++)
    {
        for (int z = 0; z < COL; z++)
        {
            if (board[k][z] != ' ')
                return 'C';
        }
    }
    return 'Q';
}
