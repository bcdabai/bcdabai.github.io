折半查找（递归做法）

#include<bits/stdc++.h>
using namespace std;

int BinFind(vector<int>v,int x,int low,int high)
{
	if(high<low)   //  尾下标小于头下标，说明没找到，返回-1 
	return -1;  
	int mid=(low+high)/2;    // 取中点 
	if(v[mid]==x)  // 如果找到，返回所在下标 
	return mid;  
	else if(x<v[mid])   // 如果要查找的元素小于中点元素，继续查找数组前半部分 
	{ 
		return BinFind(v,x,low,mid-1);
	} 
	else   // 如果要查找的元素大于中点元素，继续查找数组后半部分 
	{  
		return BinFind(v,x,mid+1,high);	
	} 
} 

int main()
{	
	vector<int>v;
	int n,m;
	cin>>n;  // 输入数组元素个数， 
	for(int i=0;i<n;i++)
	{
		cin>>m;   // 输入元素，要按照从小到大的顺序输入 
		v.push_back(m);	
	} 
	int x;
	cin>>x; // 输入要查找的元素
	int idx=BinFind(v,x,0,v.size()-1); 
	cout<<idx<<endl;   // 输出要查找的元素在数组中的下标 
	return 0;
}
