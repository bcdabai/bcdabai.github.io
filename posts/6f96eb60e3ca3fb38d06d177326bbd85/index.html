class Solution {
	public List<Integer> spiralOrder(int[][] matrix)
	{
		if (matrix.length == 0)
			return new ArrayList<Integer>();
		int l = 0;// å·¦è¾¹ç•Œ
		int r = matrix[0].length - 1;// å³è¾¹ç•Œ
		int t = 0;// ä¸Šè¾¹ç•Œ
		int b = matrix.length - 1;// ä¸‹è¾¹ç•Œ

		List<Integer> res = new ArrayList<>();
		while (true)
		{
			for (int i = l; i <= r; i++)//é¡¶éƒ¨ ä»å·¦åˆ°å³
				res.add(matrix[t][i]);
			if (++t > b)
				break;
			for (int i = t; i <= b; i++)//å³è¾¹ ä»ä¸Šåˆ°ä¸‹
				res.add(matrix[i][r]);
			if (l > --r)
				break;
			for (int i = r; i >= l; i--)//åº•éƒ¨ ä»å³åˆ°å·¦
				res.add(matrix[b][i]);
			if (t > --b)

			for (int i = b; i >= t; i--)//å·¦è¾¹ ä»ä¸‹åˆ°ä¸Š
				res.add(matrix[i][l]);
			if (++l > r)
				break;
		}
		return res;
	}
}
