输入一个整数n和n个整数，输出这n个整数的最大值。

#include<stdio.h>
int main()
{
int i;//循环变量
int n;//循环的次数
int num;//输入的值
int max;//最大的数
scanf("%d",&n);

for(i = 0;i < n;i ++)
{
scanf("%d",&num);
if(i == 0)
{
	max = num;
}
if(max < num)
{
max = num;

}

};
printf("%d",max);

return 0;

}
