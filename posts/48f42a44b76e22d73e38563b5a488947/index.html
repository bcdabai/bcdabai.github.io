利用Vue Grid Layout实现div可拖动布局

首先预览效果
注：所有的div模块都可以随意拖动，颜色样式可以在css里面
直接改

插件安装

1.npm install vue grid插件。

npm install vue-grid-layout --save

代码（包含css）

// An highlighted block
<template>
  <div style="width: 100%; height: 2000px">
    <hr />
    <br />
    <div class="container" style="width: 100%; margin-top: 10px; height: 100%">
      <grid-layout
        :layout.sync="layout"
        :col-num="12"
        :row-height="30"
        :is-draggable="draggable"
        :is-resizable="resizable"
        :responsive="responsive"
        :vertical-compact="true"
        :use-css-transforms="true"
      >
        <grid-item
          v-for="item in layout"
          :static="item.static"
          :x="item.x"
          :y="item.y"
          :w="item.w"
          :h="item.h"
          :i="item.i"
        >
          <span class="text">{{ item.i }}</span>
        </grid-item>
      </grid-layout>
    </div>
  </div>
</template>

<script>
import { GridLayout, GridItem } from "vue-grid-layout";
export default {
  components: {
    GridLayout,
    GridItem,
  },
  data() {
    return {
      layout: [
        { x: 0, y: 0, w: 2, h: 5, i: "0" },
        { x: 2, y: 0, w: 2, h: 5, i: "1" },
        { x: 4, y: 0, w: 2, h: 5, i: "2" },
        { x: 6, y: 0, w: 2, h: 5, i: "3" },
        { x: 8, y: 0, w: 2, h: 5, i: "4" },
        { x: 10, y: 0, w: 2, h: 5, i: "5" },
        { x: 0, y: 5, w: 2, h: 5, i: "6" },
        { x: 2, y: 5, w: 2, h: 5, i: "7" },
      ],
      draggable: true,
      resizable: true,
      responsive: true,
      index: 0,
    };
  },
  methods: {},
};
</script>
