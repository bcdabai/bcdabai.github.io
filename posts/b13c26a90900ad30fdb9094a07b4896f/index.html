import csv
def readcsv():
 #        请在此处添加代码         #
 # *************begin************#
    fc=open('book.csv','r',encoding='utf-8')
    lst=[]
    for line in fc:
        line = line.replace('\n','')
        lst.append(line.split(','))
    for i in range (1,len(lst)):
        print(lst[i])
    fc.close()
 # **************end*************#



if __name__ == '__main__':
    readcsv()
