去除注释代码如下：/**

* Created by PhpStorm.

* User: 25754

* Date: 2019/10/17

* Time: 9:54

*/

function removeComment($content)

{

return preg_replace("/(\/\*(\s|.)*?\*\/)|(\/\/.(\s|.*))|(#(\s*)?(.*))/", '', str_replace(array("\r\n", "\r"), "\n", $content));

}

$content = file_get_contents("./test.php");

echo removeComment($content);
