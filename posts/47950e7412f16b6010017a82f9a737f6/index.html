冒泡排序

import math
from random import random
zy=[]
for i in range(0,10):
    zy.append(int(random()*100))
print(zy)
for j in range(0,len(zy)-1) :#总共比较的轮数
    flag=1
    for i in range(0,len(zy)-1-j):#每轮比较的次数
             if zy[i] > zy[i +1]:
                tmp=zy[i]
                zy[i]=zy[i+1]
                zy[i+1]=tmp
                flag=0
    if(flag):
        print(f"经过{j+1}轮比较，已经有序")
        break
print(zy)
