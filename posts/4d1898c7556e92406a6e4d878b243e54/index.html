1.编写程序，计算序列 1 + 1/2 + 1/3 + … 的前N项之和

#include <stdio.h>
#include <math.h>
int main()
{
    double sum=0;
    int N,i=1;
    scanf("%d",&N);
    for(i=1;i<=N;i++)
    {
        sum = sum + 1.0/i;
    }
    printf("sum = %.6f",sum);
    return 0;
}

#include <math.h>
int main()
{
    int N,i=1;
    double sum=0;
    scanf("%d",&N);
    for(i=1;i<=3*N;i=i+3)

       sum = sum + pow(-1,i+1)*1.0/i*1.0;
