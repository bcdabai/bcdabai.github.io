class Solution 
{
public:
    int longestConsecutive(vector<int>& nums)
    {
        unordered_map<int , int>Hash;
        int res = 0;
        for(int i = 0; i < nums.size(); i++) 
        {
            int now = nums[i];
            if(!Hash[now]) 
            {
                int left = Hash[now - 1] , right = Hash[now + 1];
                int len = left + right + 1;
                res = max(res , len);
                Hash[now] = len;
                Hash[now - left] = len , Hash[now + right] = len;
            }
        }

        return res;
    }
};
