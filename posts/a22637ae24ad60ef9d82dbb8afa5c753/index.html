C语言 循环结构打印*号三角形

编写程序，使用循环结构语句打印输出如下图形

    *
   ***
  *****
 *******

#include<stdio.h>
int main() {
	int i,j;
	for(i=0; i<4; i++) {
		for(j=4-i; j>0; j--) {
			printf(" ");
		}
		for(j=1; j<=i*2+1; j++) {
			printf("*");
		}
		printf("\n");
	}
}
