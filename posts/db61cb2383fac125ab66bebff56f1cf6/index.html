信息学奥赛C++语言： 蛇形方阵1

【题目描述】
输入一个正整数 n，生成一个 n×n 的蛇形方阵（具体见样例）。

【输入】
一行一个正整数 n，1≤n≤20。

【输出】
共 n 行，每行 n 个正整数，每个正整数占 5 列。

【输入样例】
5

【输出样例】

代码

#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,j,i,k,t=0,a[21][21];
	cin>>n;
	for(k=1;k<=n;k++)
		if(k%2)
			for(j=1;j<=k;j++){
				i=k+1-j;
				t++;
				a[i][j]=t;
				a[n+1-i][n+1-j]=n*n+1-t;
			}
		else
			for(j=k;j>=1;j--){
				i=k+1-j;
				t++;
				a[i][j]=t;
				a[n+1-i][n+1-j]=n*n+1-t;
			}
	for(i=1;i<=n;i++){
		for(j=1;j<=n;j++)
		cout<<setw(5)<<a[i][j];
		cout<<endl;
	}
	return 0;
}
