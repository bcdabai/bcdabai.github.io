  2. import java.io.File;

  18. import com.jcraft.jsch.JSch;

  20. import com.jcraft.jsch.Session;
  21. import com.jcraft.jsch.SftpException;
  22. /**
  23. * 类说明 sftp工具类
  24. */
  25. public class SFTPUtil {
  26. private transient Logger log = LoggerFactory.getLogger( this.getClass());
  28. private ChannelSftp sftp;
  30. private Session session;
  31. /** SFTP 登录用户名*/
  32. private String username;
  33. /** SFTP 登录密码*/
  34. private String password;
  35. /** 私钥 */
  36. private String privateKey;
  37. /** SFTP 服务器地址IP地址*/
  38. private String host;
  39. /** SFTP 端口*/
  40. private int port;
  43. /**
  44. * 构造基于密码认证的sftp对象
  45. */
  46. public SFTPUtil(String username, String password, String host, int port) {
  47. this.username = username;
  48. this.password = password;
  49. this.host = host;
  50. this.port = port;
  51. }
  53. /**
  54. * 构造基于秘钥认证的sftp对象
  55. */
  56. public SFTPUtil(String username, String host, int port, String privateKey) {
  57. this.username = username;
  58. this.host = host;
  59. this.port = port;
  60. this.privateKey = privateKey;
  61. }
  63. public SFTPUtil(){}
  66. /**
  67. * 连接sftp服务器
  68. */
  69. public void login(){
  70. try {
  71. JSch jsch = new JSch();
  72. if (privateKey != null) {
  73. jsch.addIdentity(privateKey); // 设置私钥
  74. }
  76. session = jsch.getSession(username, host, port);
  78. if (password != null) {
  79. session.setPassword(password);
  80. }
  81. Properties config = new Properties();
  82. config.put( "StrictHostKeyChecking", "no");
  84. session.setConfig(config);
  85. session.connect();
  87. Channel channel = session.openChannel( "sftp");
  88. channel.connect();
  90. sftp = (ChannelSftp) channel;
  91. } catch (JSchException e) {
  92. e.printStackTrace();
  93. }
  94. }
  96. /**
  97. * 关闭连接 server
  98. */
  99. public void logout(){
  100. if (sftp != null) {
  101. if (sftp.isConnected()) {
  102. sftp.disconnect();
  103. }
  104. }
  105. if (session != null) {
  106. if (session.isConnected()) {
  107. session.disconnect();
  108. }
  109. }
  110. }
  113. /**
  114. * 将输入流的数据上传到sftp作为文件。文件完整路径=basePath+directory
  115. * @param basePath 服务器的基础路径
  116. * @param directory 上传到该目录
  117. * @param sftpFileName sftp端文件名
  118. * @param in 输入流
  119. */
  120. public void upload(String basePath,String directory, String sftpFileName, InputStream input) throws SftpException{
  121. try {
  122. sftp.cd(basePath);
  123. sftp.cd(directory);
  124. } catch (SftpException e) {
  125. //目录不存在，则创建文件夹
  126. String [] dirs=directory.split( "/");
  127. String tempPath=basePath;
  128. for(String dir:dirs){
  129. if( null== dir || "".equals(dir)) continue;
  130. tempPath+= "/"+dir;
  131. try{
  132. sftp.cd(tempPath);
  133. } catch(SftpException ex){
  134. sftp.mkdir(tempPath);
  135. sftp.cd(tempPath);
  136. }
  137. }
  138. }
  139. sftp.put(input, sftpFileName); //上传文件
  140. }
  143. /**
  144. * 下载文件。
  145. * @param directory 下载目录
  146. * @param downloadFile 下载的文件
  147. * @param saveFile 存在本地的路径
  148. */
  149. public void download(String directory, String downloadFile, String saveFile) throws SftpException, FileNotFoundException{
  150. if (directory != null && ! "".equals(directory)) {
  151. sftp.cd(directory);
  152. }
  153. File file = new File(saveFile);
  154. sftp.get(downloadFile, new FileOutputStream(file));
  155. }
  157. /**
  158. * 下载文件
  159. * @param directory 下载目录
  160. * @param downloadFile 下载的文件名
  161. * @return 字节数组
  162. */
  163. public byte[] download(String directory, String downloadFile) throws SftpException, IOException{
  164. if (directory != null && ! "".equals(directory)) {
  165. sftp.cd(directory);
  166. }
  167. InputStream is = sftp.get(downloadFile);
  169. byte[] fileData = IOUtils.toByteArray(is);
  171. return fileData;
  172. }
  175. /**
  176. * 删除文件
  177. * @param directory 要删除文件所在目录
  178. * @param deleteFile 要删除的文件
  179. */
  180. public void delete(String directory, String deleteFile) throws SftpException{
  181. sftp.cd(directory);
  182. sftp.rm(deleteFile);
  183. }
  186. /**
  187. * 列出目录下的文件
  188. * @param directory 要列出的目录
  189. * @param sftp
  190. */
  191. public Vector<?> listFiles(String directory) throws SftpException {
  192. return sftp.ls(directory);
  193. }
  195. //上传文件测试

  197. SFTPUtil sftp = new SFTPUtil( "用户名", "密码", "ip地址", 22);
  198. sftp.login();
  199. File file = new File( "D:\\图片\\t0124dd095ceb042322.jpg");
  200. InputStream is = new FileInputStream(file);
  202. sftp.upload( "基础路径", "文件路径", "test_sftp.jpg", is);

  204. }
  205. }


二、Java实现对FTP服务器的文件的上传下载

  1. import java.io.File;
  2. import java.io.FileInputStream;
  3. import java.io.FileNotFoundException;
  4. import java.io.FileOutputStream;
  5. import java.io.IOException;
  6. import java.io.InputStream;
  7. import java.io.OutputStream;
  9. import org.apache.commons.net.ftp.FTP;
  10. import org.apache.commons.net.ftp.FTPClient;
  11. import org.apache.commons.net.ftp.FTPFile;
  12. import org.apache.commons.net.ftp.FTPReply;
  14. /**
  15. * ftp上传下载工具类
  16. */
  17. public class FtpUtil {
  19. /**
  20. * Description: 向FTP服务器上传文件
  21. * @param host FTP服务器hostname
  22. * @param port FTP服务器端口
  23. * @param username FTP登录账号
  24. * @param password FTP登录密码
  25. * @param basePath FTP服务器基础目录
  26. * @param filePath FTP服务器文件存放路径。文件的路径为basePath+filePath
  27. * @param filename 上传到FTP服务器上的文件名
  28. * @param input 输入流
  29. * @return 成功返回true，否则返回false
  30. */
  31. public static boolean uploadFile(String host, int port, String username, String password, String basePath,
  32. String filePath, String filename, InputStream input) {
  33. boolean result = false;
  34. FTPClient ftp = new FTPClient();
  35. try {
  36. int reply;
  37. ftp.connect(host, port); // 连接FTP服务器
  38. // 如果采用默认端口，可以使用ftp.connect(host)的方式直接连接FTP服务器
  39. ftp.login(username, password); // 登录
  40. reply = ftp.getReplyCode();
  41. if (!FTPReply.isPositiveCompletion(reply)) {
  42. ftp.disconnect();
  43. return result;
  44. }
  45. //切换到上传目录
  46. if (!ftp.changeWorkingDirectory(basePath+filePath)) {
  47. //如果目录不存在创建目录
  48. String[] dirs = filePath.split( "/");
  49. String tempPath = basePath;
  50. for (String dir : dirs) {
  51. if ( null == dir || "".equals(dir)) continue;
  52. tempPath += "/" + dir;
  53. if (!ftp.changeWorkingDirectory(tempPath)) { //进不去目录，说明该目录不存在
  54. if (!ftp.makeDirectory(tempPath)) { //创建目录
  55. //如果创建文件目录失败，则返回
  56. System.out.println( "创建文件目录"+tempPath+ "失败");
  57. return result;
  58. } else {
  59. //目录存在，则直接进入该目录
  60. ftp.changeWorkingDirectory(tempPath);
  61. }
  62. }
  63. }
  64. }
  65. //设置上传文件的类型为二进制类型
  66. ftp.setFileType(FTP.BINARY_FILE_TYPE);
  67. //上传文件
  68. if (!ftp.storeFile(filename, input)) {
  69. return result;
  70. }
  71. input.close();
  72. ftp.logout();
  73. result = true;
  74. } catch (IOException e) {
  75. e.printStackTrace();
  76. } finally {
  77. if (ftp.isConnected()) {
  78. try {
  79. ftp.disconnect();
  80. } catch (IOException ioe) {
  81. }
  82. }
  83. }
  84. return result;
  85. }
  87. /**
  88. * Description: 从FTP服务器下载文件
  89. * @param host FTP服务器hostname
  90. * @param port FTP服务器端口
  91. * @param username FTP登录账号
  92. * @param password FTP登录密码
  93. * @param remotePath FTP服务器上的相对路径
  94. * @param fileName 要下载的文件名
  95. * @param localPath 下载后保存到本地的路径
  96. * @return
  97. */
  98. public static boolean downloadFile(String host, int port, String username, String password, String remotePath,
  99. String fileName, String localPath) {
  100. boolean result = false;
  101. FTPClient ftp = new FTPClient();
  102. try {
  103. int reply;
  104. ftp.connect(host, port);
  105. // 如果采用默认端口，可以使用ftp.connect(host)的方式直接连接FTP服务器
  106. ftp.login(username, password); // 登录
  107. reply = ftp.getReplyCode();
  108. if (!FTPReply.isPositiveCompletion(reply)) {
  109. ftp.disconnect();
  110. return result;
  111. }
  112. ftp.changeWorkingDirectory(remotePath); // 转移到FTP服务器目录
  113. FTPFile[] fs = ftp.listFiles();
  114. for (FTPFile ff : fs) {
  115. if (ff.getName().equals(fileName)) {
  116. File localFile = new File(localPath + "/" + ff.getName());
  118. OutputStream is = new FileOutputStream(localFile);
  119. ftp.retrieveFile(ff.getName(), is);
  120. is.close();
  121. }
  122. }
  124. ftp.logout();
  125. result = true;
  126. } catch (IOException e) {
  127. e.printStackTrace();
  128. } finally {
  129. if (ftp.isConnected()) {
  130. try {
  131. ftp.disconnect();
  132. } catch (IOException ioe) {
  133. }
  134. }
  135. }
  136. return result;
  137. }
  139. //ftp上传文件测试main函数
  140. public static void main(String[] args) {
  141. try {
  142. FileInputStream in= new FileInputStream( new File( "D:\\Tomcat 5.5\\pictures\\t0176ee418172932841.jpg"));
  143. boolean flag = uploadFile( "192.168.111.128", 21, "用户名", "密码", "/www/images", "/2017/11/19", "hello.jpg", in);
  144. System.out.println(flag);
  145. } catch (FileNotFoundException e) {
  146. e.printStackTrace();
  147. }
  148. }
  149. }
