RichTextBox获取当前行、列，统计行数

此代码根据网上代码更改

//当前光标所在位置
            //当前行的索引
            int index = this.richTxt.GetFirstCharIndexOfCurrentLine();
            //共有行数
            int intLines = this.richTxt.GetLineFromCharIndex(this.richTxt.TextLength) + 1;
            //得到光标的行号
            int line = this.richTxt.GetLineFromCharIndex(index) + 1;
            //得到光标列的索引  
            /*SelectionStart得到光标所在位置的索引 
                再减去 
                当前行第一个字符的索引 
                = 光标所在的列数(从0开始）
            */
            int column = this.richTxt.SelectionStart - index + 1;
            //this.richTxt.Paste(Clipboard.GetDataObject());
            this.toolStripStatusLabel_Row.Text = string.Format("第{0}行{1}列,共{2}行", line.ToString(), column.ToString(), intLines.ToString());
