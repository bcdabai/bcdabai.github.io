js中时间戳转换成时间

function timestampToTime(timestamp) {
                var date = new Date(timestamp );//时间戳为10位需*1000，时间戳为13位的话不需乘1000
                Y = date.getFullYear() + '-';
                M = (date.getMonth()+1 < 10 ? '0'+(date.getMonth()+1) : date.getMonth()+1) + '-';
                D = date.getDate() + ' ';
                h = date.getHours() + ':';
                m = date.getMinutes() + ':';
                s = date.getSeconds();
                return Y+M+D;//时分秒可以根据自己的需求加上
            }

运用上面输出的时间格式为：yyyy-MM-dd
