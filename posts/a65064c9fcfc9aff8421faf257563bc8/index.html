string op(string now, char c){
	string ans = "";
	if(c == 'A'){
		for(int i = 7; i >= 4; i--) ans += now[i];
		for(int i = 3; i >= 0; i--) ans += now[i];
	}
	else if(c == 'B'){
		ans += now[3];
		for(int i = 0; i < 3; i++) ans += now[i];
		for(int i = 5; i <= 7; i++) ans += now[i];
		ans += now[4];
	}
	else{
		int gs[] = {0, 6, 1, 3, 4, 2, 5, 7};
		for(int i = 0; i < 8; i++) ans += now[gs[i]];
	} 
	return ans;
}

int bfs(){
	int hh = 0, tt = -1;
	
	q[++tt] = st;
	dis[st] = 0;
