C语言，输入三个数，求出最大数，最小数并输出

  1. 使用if进行解答 
    #include<stdio.h>
void main()
{
	int a,b,c,t,q;
	printf("please input a,b,c");
	scanf("%d,%d,%d",&a,&b,&c);
	if(a>=b&&a>=c) t=a;
	else if(b>=a&&b>=c) t=b;
	else t=c;
	if(a<=b&&a<=c) q=a;
	else if(b<=a&&b<=c) q=b;
	else q=c;
	printf("max:%d\nmin:%d",t,q);
}
  2. 使用三目运算进行解答  
    #include<stdio.h>
void main()
{
	int a,b,c,t,q;
	printf("please input a,b,c");
	scanf("%d,%d,%d",&a,&b,&c);
    t=a>=b?a:b;
    t=t>=c?t:c;
    q=a<=b?a:b;
    q=q<=c?q:c; 
	printf("max:%d\nmin:%d",t,q);
}
