import os,sys

cmd = 'cdrive download '

def download(md_dir):
    cwd = os.getcwd()

    old_root = md_dir

    for root,dirs,files in os.walk(md_dir):
        for file in files:
            if root!=old_root:
                old_root = root
                new_dir = cwd+'/'+(root[len(md_dir):])
                
                try:
                    os.mkdir(new_dir)
                finally:
                    print('dir exist %s'%new_dir)

                os.chdir(new_dir)
                print('----- deal '+root)
            file_obj = open(root+'/'+file,'r')
            try:
                for line in file_obj:
                    for item in line.split('|'):
                        litem = item.lstrip()
                        if litem.startswith('''bdrive://''') or litem.startswith('bdex://') or litem.startswith('bjdrive://') \
                            or litem.startswith('jsdrive://') or litem.startswith('csdrive://') or litem.startswith('wbdrive://'):                        
                            os.system(cmd+litem)
            finally:
                file_obj.close()

if __name__=='__main__':
    if len(sys.argv) != 2:
        print('please with md file dir')
    else:
        download(sys.argv[1])
