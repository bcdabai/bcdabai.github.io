I. Modulo Permutations

  • 题目大意：给出 n n n，求在 n ! n! n!种排列中，有多少种排列满足所有的 i i i满足 p i + 1 % p i < = 2 p_{i+1}\%p_i<=2 pi+1​%pi​<=2(环形排列)
  • n < = 1 e 6 n<=1e6 n<=1e6

观察发现：可以把数组分为两段，一段从1开始，一段从2开始。每一段都要满足后一个数%前一个数<=2
例如：n=8的一种情况: [ 1 , 3 , 5 , 7 ] , [ 2 , 4 , 6 , 8 ] [1,3,5,7],[2,4,6,8] [1,3,5,7],[2,4,6,8]，合并为 [ 1 , 3 , 5 , 7 , 2 , 4 , 6 , 8 ] [1,3,5,7,2,4,6,8] [1,3,5,7,2,4,6,8]即为合法答案

然后从1开始和从2开始的合并就是答案了。

考虑使用 d p [ a ] [ b ] dp[a][b] dp[a][b]表示第一个数组以a结尾，第二个数组以b结尾的方案数。
发现我们可以一段一段转移，就是只从 d p [ a ] [ a − 1 ] dp[a][a-1] dp[a][a−1]开始转移。
于是使用 d p [ x ] dp[x] dp[x]表示一段为x结尾，另一段以x-1结尾

然后转移可以从 d p [ x ] dp[x] dp[x]转移至 d p [ k ∗ ( x − 1 ) ] dp[k*(x-1)] dp[k∗(x−1)]至 d p [ k ∗ ( x − 1 ) + 2 ] dp[k*(x-1)+2] dp[k∗(x−1)+2](k为整数)
然后由于是个调和级数的样子，所以时间复杂度是 O ( n ∗ l n n ) O(n*lnn) O(n∗lnn)

最后答案是 n ∗ ∑ i d p [ i ] n*\sum_i{dp[i]} n∗∑i​dp[i]，因为一开始有n种开始的地方。

代码实现如下：

#include<bits/stdc++.h>
#define ll long long
#define maxn 1000005
#define rep(i,a,b) for(int i=a;i<=b;i++)
using namespace std;

ll n,dp[maxn];
const ll mod=1000000007;
unordered_map <int,int> p;

int main()
{
	cin>>n; if(n<=2) {printf("%d\n",n); return 0;}
	dp[2]=1; dp[3]=1; dp[4]=2;
	rep(i,2,n)
	{
		if(i<=5)
		{
			p.clear();
			for(int j=i-1;j<=n;j+=(i-1)) rep(k,0,2)
				if(j+k>i&&!p[j+k]) dp[j+k]=(dp[j+k]+dp[i])%mod,p[j+k]=1;
		}
		else
		{
			for(int j=i-1;j<=n;j+=(i-1)) rep(k,0,2)
				if(j+k>i) dp[j+k]=(dp[j+k]+dp[i])%mod;
		}
	}
	rep(i,1,n-1) dp[n]=(dp[n]+dp[i])%mod; cout<<(dp[n]*n)%mod<<endl;
	return 0;
}

L. Neo-Robin Hood

  • 题目大意：有n个人，你一开始有0元，每个人有两种权值 m [ i ] , p [ i ] m[i],p[i] m[i],p[i]，你可以选择从中选k个人抢掉 m [ i ] m[i] m[i]元，再选另k个人给他 p [ i ] p[i] p[i]元，要求自己的最终钱数大于0，求能选出的最大的k是多少。
  • n < = 1 e 5 , m [ i ] , p [ i ] < = 1 e 9 n<=1e5,m[i],p[i]<=1e9 n<=1e5,m[i],p[i]<=1e9

首先考虑：
选出了一个大小为2*k的集合，从中抢钱/给出钱
如果抢了a，给了b钱 和 抢了b的钱，给a哪种方案更优秀。
如果抢a更加优秀的话，则有：
m [ a ] − p [ b ] > m [ b ] − p [ a ] m[a]-p[b]>m[b]-p[a] m[a]−p[b]>m[b]−p[a]
变形得： m [ a ] + p [ a ] > m [ b ] + p [ b ] m[a]+p[a]>m[b]+p[b] m[a]+p[a]>m[b]+p[b]

故按照 m [ i ] + p [ i ] m[i]+p[i] m[i]+p[i]排序，答案一定是选出2*k个数中抢前k个人，援助后面k个人。

所以我们枚举从 [ 1 , i ] [1,i] [1,i]选人抢， [ i + 1 , n ] [i+1,n] [i+1,n]选人援助
那么二分答案得到一个k，维护一下前缀前 k k k大的 p [ i ] p[i] p[i]，后缀前k小的 m [ i ] m[i] m[i]就做完了。

时间复杂度 O ( n ∗ l o g 2 n ) O(n*log^2n) O(n∗log2n)

代码实现如下：

#include<bits/stdc++.h>
#define ll long long
#define maxn 1000005
#define inf 1e9
#define ins insert
#define pb push_back
#define vi vector <int>
#define rep(i,a,b) for(int i=a;i<=b;i++)
#define per(i,a,b) for(int i=a;i>=b;i--)
using namespace std;

inline ll read()
{
	ll x=0,w=1; char c=getchar();
	while(c<'0'||c>'9') {if(c=='-') w=-1; c=getchar();}
	while(c<='9'&&c>='0') {x=(x<<1)+(x<<3)+c-'0'; c=getchar();}
	return w==1?x:-x;
}

struct node{int m,p,tot;}p[maxn];
inline bool cmp(node a,node b){return a.tot>b.tot;}

multiset <int> s;
multiset <int,greater<int> > s2;
multiset <int>::iterator it;
multiset <int,greater<int> >::iterator it2;

int n;
ll pre[maxn],suf[maxn];

inline bool jud(int x)
{
	ll nw=0,tmp=x; s.clear(); s2.clear();
	rep(i,1,n)
	{
		if(i<=x) s.ins(p[i].m),nw+=p[i].m;
		else
		{
			it=s.begin(); int tmp=*it;
			if(tmp<p[i].m) nw-=tmp,nw+=p[i].m,s.erase(it),s.ins(p[i].m);
		}
		pre[i]=nw;
	}
	
	nw=0;
	per(i,n,1)
	{
		if(n-i+1<=x) s2.ins(p[i].p),nw+=p[i].p;
		else
		{
			it2=s2.begin(); int tmp=*it2;
			if(tmp>p[i].p) nw-=tmp,nw+=p[i].p,s2.erase(it2),s2.ins(p[i].p);
		}
		suf[i]=nw;
	}
	rep(i,x,n-x) if(pre[i]>=suf[i+1]) return true;
	return false;
}

int main()
{
	//freopen("t1.in","r",stdin);
	n=read(); rep(i,1,n) p[i].m=read();
	rep(i,1,n) p[i].p=read(),p[i].tot=p[i].m+p[i].p; sort(p+1,p+n+1,cmp);
	int l=1,r=n/2,ans=0;
	while(l<=r)
	{
		int mid=(l+r)>>1;
		if(jud(mid)) l=mid+1,ans=mid;
		else r=mid-1;
	}
	cout<<ans<<endl;
	return 0;
}
