int reverseInt( int num ){
	int result = 0;
	// 将整数倒序
	for (int i = num; i ; )
	{
		// 去掉个位上的数
		i = i/10*10;
		// 将result进一位，并加上当前个位上的数（num-i）
		result = result * 10 + num - i;

		i = i /10;
		num = num/10;
	}
	return result;
}

// 第二种实现
int reverseInt2( int num ){	
	int result = num%10;
	for (; num/=10; )
	{
		result = result*10 + num%10;
	}
	return result;	
}

void main(){

	int num;
	scanf("%d",&num);

	// 反转整数
	int result = reverseInt(num);

	printf("result = %d",result);

	system("pause");
}
