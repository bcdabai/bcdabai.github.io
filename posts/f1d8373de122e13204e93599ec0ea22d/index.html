vue路由传值及接收传递数据

传递数据

<template>
  <div id="app">
    <router-link :to="{path: '/page', query:{page}}">router-link传值</router-link>
    <div @click='goPage()'>方法传值</div>
  </div>
</template>

JS

<script>
export default {
    data () {
       return {
         page：‘传递数据’
       }
    }

    methods:{
        goPage(){   //路由传值
           this.$router.push({name:'page',params:{msg:page}})   
        },
    }

}
</script>
