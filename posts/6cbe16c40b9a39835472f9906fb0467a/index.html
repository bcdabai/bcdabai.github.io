1-5 线性表元素的区间删除 PTA

List Delete (List L,ElementType minD,ElementType maxD)
{   
    int i=0,j=0;
    for(i=0;i<=L->Last;i++)
    {
        if (L->Data[i]>=maxD||L->Data[i]<=minD)
        {
           L->Data[j]=L->Data[i];
           j++;
        }
    }  //这里跟4有点出入，不需要else，不需要浮标长度，因为L->Last直接就可指向了长度
     L->Last=j-1;//别忘了-1，因j++多加了1
    return L;
}
