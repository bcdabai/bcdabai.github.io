C语言画爱心

首先,你要了解画爱心的循环代码:

	for (y = 1.5f; y > -1; y -= 0.1f)
	{
		for (x = -1.5f; x < 1.5f; x += .05f) {
			a = x * x + y * y - 1;
			putchar(a * a * a - x * x * y * y * y < 0.0f ? 'x' : ' ');
		}

		putchar('\n'); //换行 
	}

其次,如果要更换颜色,就要用到设计程序颜色(system("color **"))。

0~9为背景颜色设置，A~F为字体颜色设置。

例如：system（“color 0C”）就是黑色背景，淡红色字体

system(" color 0C");//设计程序颜色

切记：使用color函数是要包含头文件(Windows.h)

知道这些实现代码就很简单了：

#define _CRT_SECURE_NO_WARNINGS 1
#include<stdio.h>
#include<Windows.h>
int main()
{
	system(" color 0C");//颜色
	printf("某某某，爱你哟！！！\n");//可有可无，看自己
	float x, y, a;
	for (y = 1.5f; y > -1; y -= 0.1f)
	{
		for (x = -1.5f; x < 1.5f; x += .05f) {
			a = x * x + y * y - 1;
			putchar(a * a * a - x * x * y * y * y < 0.0f ? 'x' : ' ');
		}

		Sleep(150);//睡眠1.5s
		printf("\n");
	}
	return 0;
}
