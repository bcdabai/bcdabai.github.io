用JAVA程序计算1！+2！+3！+........+10!的和

public class 阶乘 {
//    用JAVA程序计算1！+2！+3！+........+10!的和
    public static void main(String args[]){

        int i,j,n,sum;
        sum=0;
        for(i=1; i<=10;i++){
            n=1;
            for(j=1;j<=i;j++){
               n=n*j;
            }
            sum=sum+n;
        }
         System.out.print("1!+2!=3!+............+10!="+sum) ;

    }



}
