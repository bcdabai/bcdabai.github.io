封装求和函数

//封装一个求和函数
        function sum(arr){
            var total = 0;
            for (var i=0;i<arr.length;i++){
                total += arr[i];
            }
            return total;
        }
        var numbers = [-10,-2222,-35,-4];
        console.log(sum(numbers));//-2217
