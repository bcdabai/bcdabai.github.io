df_stock.set_index(‘date’,inplace=True)
df_stock=ts.get_k_data(‘300015’)
print(‘stock_1’,df_stock.loc[‘2018-05-22’:‘2018-05-25’]) ## 2018年中期分红送股后前复权数据
