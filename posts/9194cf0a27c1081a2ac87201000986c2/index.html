PHP将数组元素两两组合为新数组

// 方式一
for($i=1; $i<=4; $i++){
    $array[] = $i;
}
for($i=0; $i<count($array); $i++){
    for($j=0; $j<count($array)-$i-1; $j++) {
        $array2[] = [$array[$i], $array[$i+$j+1]];
    }
}
print_r($array2);

// 方式二
$arr = ['A', 'B', 'C', 'D'];
$new_arr = [];
while (count($arr) > 1) {
    $item = array_shift($arr);
    foreach ($arr as $value) {
        $new_arr[] = [$item, $value];
    }
}
var_dump($new_arr);

// 方式三
// 数组中任意两个元素之间都要组合一下
$r = [];
$arr = array('A','B','C','D');
$num = count($arr);
for ($i=0;$i<$num; $i++) {
    for($j=$num-1; $j>$i; $j--){
        $r[] = [$arr[$i], $arr[$j]];
    }
}
print_r($r);
