                        DecimalFormat df = new DecimalFormat("#.##");
                        value = df.format(cell.getNumericCellValue());
                    }
                    break;
                case 1:
                    value = cell.getStringCellValue();
                    break;
                case 2:
                    value = cell.getCellFormula();
                case 3:
                case 5:
                default:
                    break;
                case 4:
