设计一个递归函数，求x的n次幂

//2019-11-6
#include <stdio.h>
//设计一个递归函数，求x的n次幂：
//主函数通过键盘输入x和n的值，并输出x^n的值（假设x为实数，n为正整数）
double fun(double x,int n){
	if(n==1)
		return x;
	return x*fun(x,n-1);
}
int main()
{
	double x;
	int n;
	double result=0;
	scanf("%lf %d",&x,&n);
	result=fun(x,n);
	printf("%.2lf\n",result);
	return 0;
}
