双向链表--尾插法

typedef struct Node {
	struct Node* prev;
	int val;
	struct Node* next;
}Node, *linklist;

1.尾插法

linklist Taillist(Node* L) {
	L->next = NULL;
	L->prev = NULL;
	Node* p;
	Node* r;
	r = L;
	for (int i = 0; i < 5; i++) {
		p = (Node*)malloc(sizeof(Node));
		p->val = i + 1;
		p->prev = r;
		r->next = p;
		r = p;


	}
	r->next = NULL;
	return L;
}
