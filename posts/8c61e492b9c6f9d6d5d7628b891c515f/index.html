      shTempData: [{ name: '安徽阜阳宝龙开闭所', value: [116, 33] }]

    // console.log(JSON)

      geo: {
        map: '阜阳市', // js 地图包要和echarts.registerMap()里面的名字保持一致

        roam: true,
        zoom: 1,
        zlevel: 2,
        // data: this.shTempData, // 圆点进度纬度
        label: { // 显示地区名
          normal: {
            show: true,
			// 提示内容 里面渲染的是data里面的数据

              return params.name

            position: 'top', // 提示方向
            color: '#fff'

          emphasis: {
            show: true // 点
          }
        },

            color: 'orangered', // 圆球拨动的颜色

            borderWidth: 0.5,
            areaColor: '#021642'

          emphasis: {
            areaColor: 'orangered', // 滑过高亮颜色
            borderColor: '#111'
          }
          // areaColor: '#021642'

        nameMap: {
          阜南县: '阜南县', // 在画图时候起的名字和data的name这个对应
          界首市: '界首市',
          临泉县: '临泉县',

          颍东区: '颍东区',
          颍泉区: '颍泉区',

        roam: true

        trigger: 'item',

        //   return data.data.name + '异常人数：' + data.data.valuea

      // 工具箱
      //   toolbox: {
      //     show: true,
      //     orient: 'vertical',
      //     left: 'right',

      //     realtime: false,
      //     calculable: true,
      //     inRange: {
      //       color: ['green', 'red']
      //     }
      //   },
      series: [
        {
          name: ' ',
          type: 'effectScatter', // 涟漪动画
          mapType: '阜阳市', // 自定义扩展图表类型
          coordinateSystem: 'geo',
          roam: true,
          zoom: 1,
          zlevel: 2,

          symbolSize: 15, // 标记的大小。

            period: 4, // 动画速度，值越小，动画越快

          label: {
            normal: {
              show: true,
              // 提示内容

            }

            normal: {

              areaColor: '#021642'

            阜南县: '阜南县', // 在画图时候起的名字和data的name这个对应

          }
        }
      ]

  }
}
