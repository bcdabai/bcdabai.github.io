一、定义奖牌信息结构体

typedef struct LS{
	char gjm[30];
	int jp, yp, tp, jpzs, zpm;
}ls;

二、从本地文件中读取奖牌文件

《冬奥会奖牌榜.txt》此文件需要自己放入文件夹中，作为最初文件！！

int Reading(ls* ip,int len){
	char buffer[200];
	FILE *fp = fopen("冬奥会奖牌榜.txt","r"); // 打开文件
	int i=0;
	if(fp == NULL){
		perror("fp");
		return -1;
	}
	fgets(buffer, 199, fp);
	for(i=0;i<len;i++){
		fscanf(fp,"%s%d%d%d%d%d",
			ip[i].gjm, &ip[i].jp, &ip[i].yp, &ip[i].tp, &ip[i].jpzs, &ip[i].zpm);
		if(feof(fp)) break;// 如果文件结束了，也退出循环 
	}
	fclose(fp);
	fp = NULL;
	return i+1;
}

四、对奖牌榜排序

a、交换函数

void Swap(ls* a, ls *b) {
	ls temp = *a;
	*a = *b, *b = temp;
}

b、排序

此次排序是按照金牌数从高到低，金牌数相同，则看银牌数，以此类推

void SortData(ls* ip, int len) {
	for (int i = 0; i < len; i++) {
		for (int j = i + 1; j < len; j++) {
			if (ip[i].jp < ip[j].jp)
				Swap(&ip[i], &ip[j]);
			else if (ip[i].jp == ip[j].jp) {// 金牌数相同 
				if (ip[i].yp < ip[j].yp)
					Swap(&ip[i], &ip[j]);
				else if (ip[i].yp == ip[j].yp) {// 银牌数相同 
					if (ip[i].tp < ip[j].tp)
						Swap(&ip[i], &ip[j]);
				}
			}
		}
	}
}
