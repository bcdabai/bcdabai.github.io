install git lfs on ubuntu

sudo apt-get install software-properties-common

sudo curl -s https://packagecloud.io/install/repositories/github/git-lfs/script.deb.sh > script.sh

sudo bash script.sh

sudo apt-get install git-lfs

git lfs install
