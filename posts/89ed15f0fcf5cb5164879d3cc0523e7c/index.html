docker基础使用docker常用命令

docker ps -a  ##查看所有容器

docker rm [-f] $(docker ps -qa)  ##删除所有容器

docker rmi $(docker images -q -f dangling=true)  ##删除虚悬镜像

docker build -t yym_image:v2 .  ##制作镜像

docker run -d -it --name MOP_ac_container -v /home/yym/test/yym_logs/:/home/yym/test/yym_logs/ -p 7676:7676 -p 9999:9999 yym_image:v2    ##启动容器

docker start 8814e4d2df92 ##启动容器

docker exec -it 49fc12d95ad8 /bin/bash  ##进入容器

docker restart 8814e4d2df92 ##重新启动容器

docker stop 8814e4d2df92  ##关闭容器

docker logs 12a2fe18907a ##查看容器日志

docker inspect 12a2fe18907a ##查看容器信息

docker network ls ##查看网络

docker inspect network 859ae1bc36af  ##查看容器网络

docker top 04ed3370599b ##查看容器进程

docker stats 04ed3370599b ##查看容器资源

docker port e778fd59f3f9 ##查看容器端口

docker pull nginx:1.21.3 ##docker拉取nginx1.21.3镜像

docker history nginx:1.21.3 ##镜像使用了哪些 image layer

docker cp outservice_nginx_proxy:/etc/nginx/nginx.conf /home/k8s/nginx_conf  ##拷贝容器文件到外部

docker cp /home/k8s/nginx_conf/default.conf outservice_nginx_proxy:/etc/nginx/conf.d/  ##外部文件拷贝到容器里面

docker run -it --net container:04ed3370599b --name testbusybox busybox  ##新启动容器使用04ed3370599b容器网络

docker network connect my-net my-nginx ##容器使用自定义网络 容器增加网卡

docker search oceanbase  ##docker搜索镜像

docker images  ##docker查看所有镜像

docker image inspect 3fa112fd3642 ##查看镜像分层
