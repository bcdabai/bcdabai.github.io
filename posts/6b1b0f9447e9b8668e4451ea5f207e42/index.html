编写程序，根据公式e=1+1/1!+1/2！….+1/n!；计算e的近似值，直到最后一项小于10^-5为止

本题需要使用for循环语句将多个1/n！相加

#include <iostream>
using namespace std;

int main( )
{
	double sum = 0;
	double i = 1.0;
	int n;

	for (n = 1; i >= 0.00001; n++)
	{
		sum =sum + i;
		i =i/ n;
	}

	cout << "当n为 " << n - 1 << " 时最后一项小于0.00001" << endl;
	cout << "e的值为 " << sum;

	return 0;
}
