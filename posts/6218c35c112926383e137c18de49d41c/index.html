public class Test15 {
    public static void main(String[] args) {

        int[] arr = {2023,0401,1100};

        String s = arrayToString(arr);

        System.out.println("订单号："+s);
    }

    private static String arrayToString(int[] arr) {

        String str = "[";
        for(int i = 0;i < arr.length;i++){
            str += arr[i];
        }
        str +="]";
        return str;
    }

}
