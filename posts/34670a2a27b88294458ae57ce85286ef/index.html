10.4：输入两个正整数m和n，求其最大公约数和最小公倍数

#include <head.h>
#include <stdio.h>

int gcd(int m, int n) 
{
    if (n == 0) 
    {
        return m;
    }
    return gcd(n, m % n);
}

int lcm(int m, int n) 
{
    return m * n / gcd(m, n);
}

int main(int argc, const char *argv[]) 
{
    int m, n;
    printf("Enter two positive integers: ");
    scanf("%d %d", &m, &n);
    printf("GCD: %d\n", gcd(m, n));
    printf("LCM: %d\n", lcm(m, n));
    return 0;
}
