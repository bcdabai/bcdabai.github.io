编写一个函数，用于去掉字符串前面和后面的空格

编写一个函数，用于去掉字符串前面和后面的空格

例如输入" I am a student “，输出"I am a student”

#include <iostream>
#include<cstring>
using namespace std;
void mytrim_1(char *string)//删除字符串前面的空格
{
    char *p = string;
    while(*p == ' ')
    {
        p++;           //跳过前面的空格
    }
    while(*p !='\0')
    {
        *string = *p; //然后*p从不是空格开始，进行字符串拷贝，
        string++;
        p++;
    }
    *string = '\0';
}
void mytrim_2(char *string)//删除字符串后面的空格
{
    int i;
    int len = strlen(string) ;
    for(i = len-1;i>=0;i--)
    {
        if (*(string + i) == ' ')
            *(string + i) = '\0';  
        else
            break;
    }
}
int main()
{
    char str[100] = "   I am a student!    ";
    mytrim_1(str);
    mytrim_2(str);
    cout<<str<<"“没空格啦”"<<endl; 
 return 0;
}
