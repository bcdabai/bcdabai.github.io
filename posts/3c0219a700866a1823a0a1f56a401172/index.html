VUE 钉钉扫码接口IOS设备报错

vue开发的钉钉h5微应用IOS使用钉钉的扫码接口时报错

调用时将type: String,为了省事直接改为’all’

const ScanQrCode = () => {
  dd.biz.util.scan({
    type: 'all', // type 为 all、qrCode、barCode，默认是String。iPhone使用'all'才能正常使用
    onSuccess: function (data: any) {
      //onSuccess将在扫码成功之后回调
      /* data结构
        { 'text': String}
      */
      console.log('扫码获取到设备编号',data)
    },
    onFail: function (err: any) {
      console.log('扫码失败')
    }
  })
}
