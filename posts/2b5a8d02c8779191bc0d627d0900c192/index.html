数组根据某字段进行分组

<?php
// 数据$arr从别人哪里copy的，废话不多说，拿去运行吧
$arr = Array(
    0 => Array
    (
        'title' => '新闻投票分成',
        'gold' => +0.004095,
                            'create_time' => 1545355432,
                            'create_date' => 20181221,
                        ),

                    1 => Array
(
    'title' => '新闻投票分成',
    'gold' => +0.002801,
                            'create_time' => 1545355432,
                            'create_date' => 20181221,
                        ),

                    2 => Array
(
    'title' => '持币分红',
    'gold' => +0.837871,
                            'create_time' => 1545194334,
                            'create_date' => 20181219,
                        )

                );


// 按照日期进行分组

function arrayGroup($arr)
{
    foreach ($arr as $key => $value) {
        $newArr[$value['create_date']][] = $value;
    }

    return $newArr;
}
