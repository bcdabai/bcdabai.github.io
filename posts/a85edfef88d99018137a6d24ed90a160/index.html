  1.比较两个数大小

 

 

    MOV AX,DATAS
    MOV DS,AX

	mov al,5
	cmp al,9
	ja L1        ;ja，若大于跳转至L1
	
	mov dl,'N'    ;若没有jmp，则继续程序
	mov ah,2
	int 21h
	jmp L2
	
L1:
	mov dl,'Y'
	mov ah,2
	int 21h
	
L2:

    MOV AH,4CH
    INT 21H

  2.计算累加和超过100

 

 

    MOV AX,DATAS
    MOV DS,AX

    mov ax,0
    mov cx,1
    
again:
	add ax,cx
	cmp ax,100
	ja over
	add cx,1
	jmp again
	
over:

    MOV AH,4CH
    INT 21H
