JS实现千位分隔符，用于处理金额

 有时候处理金额的时候，会将金额进行千分位分割，其实一个很简单的函数即可搞定。

function money_format(money) {
      if (money !== undefined) {
        const arr = money.toString().split('.');
        return (
          (arr[0] || 0).toString().replace(/(\d)(?=(?:\d{3})+$)/g, '$1,') +
          (arr[1] ? '.' + arr[1] : '')
        );
      } else {
        return money;
      }
 }

money_format(2000000) // 2,000,000
