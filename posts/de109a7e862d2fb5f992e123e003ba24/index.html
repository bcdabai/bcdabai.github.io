【案例3-2】银行存取款
对于银行存取款的流程，人们非常熟悉，用户可在银行对自己的资金账户进行存款、取款、查询余额等操作，极大地方便了人们对资金的管理。
本案例要求使用所学的知识编写一个程序，实现银行存取款功能。案例要求具体如下：
(1）创建账户，初始存款为500元。
(2）向账户存入1000元。
(3）从账户取出800元。

ps：这个案例并不难，就是练习构造方法。

 * ZAY 2023.7.19

            } else {

        }

//定义创建账户类
class User {
    public static void User() {
        Scanner sc = new Scanner(System.in);
        System.out.println("请输入六位数账号：");
        int m = sc.nextInt();
        int len = (m + "").length();//将数字转化为字符串求长度
        if (len != 6) {
            System.out.println("输入错误，请重新输入");
        }else {
            System.out.println("创建成功");
        }
    }
}

//定义存钱类
class Saving {
    static int money = 500;
    public static void Saving() {
        System.out.println("请输入您的存款金额：");
        Scanner sc = new Scanner(System.in);
        int m = sc.nextInt();
        money = money + m;
        System.out.println("当前余额为" + money);
    }
}

//定义取钱类
class Withdraw {
    public  static void Withdraw () {
        System.out.println("请输入您的取款金额：");
        Scanner sc = new Scanner(System.in);
        int m = sc.nextInt();
        Saving.money = Saving.money - m;
        if (Saving.money < 0) {
            System.out.println("金额不足");
        } else {
            System.out.println("当前余额为" + Saving.money);
        }
    }
}
