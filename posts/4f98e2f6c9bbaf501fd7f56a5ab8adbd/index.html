1到10之间所有数的平方和立方

#include <stdio.h>
main()
{
int i,n;
long p=1,j=1;
printf("please enter n:");
scanf("%d",&n);
for(i=1;i<=n;i++)
{
p=i*i;
printf("%d=%1d\n",i,p);
}

for(i=1;i<=n;i++)
{
j=i*i*i;
printf("%d=%1d\n",i,j);
}
}
