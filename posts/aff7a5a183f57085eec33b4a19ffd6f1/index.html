python，Excel表格追加数据，不覆盖原数据。

import xlwt,xlrd
import xlutils.copy
xing=1 #要追加或修改的行数的数据
filename = "每日更新" #要追加或者修改表格的文件名。
def writeadd():
			data = xlrd.open_workbook(filename+".xls")#读入表格
			ws = xlutils.copy.copy(data) #复制之前表里存在的数据
			table=ws.get_sheet(0)
			table.write(xing, 1, label="data1") #追加数据
			table.write(xing, 2, label="data2") #追加数据
			table.write(xing, 3, label="data3") #追加数据
			table.write(xing, 4, label="data4") #追加数据
			ws.save(filename +".xls")  #保存的有旧数据和新数据   
writeadd()
