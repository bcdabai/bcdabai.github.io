D.序列求和 (10分）
答案：101449

思路：
本题要求包含即可，不是恰好包含。 我们可以直接枚举1~ 10000甚至更大的数，求出它的因子个数cnt，若第一次包含1~cnt之内因子的数还没有找到，显然就为当前枚举的这个数了。

#include<bits/stdc++.h>
using namespace std;
#define INF 0x7fffffff
#define pb(x) push_back(x)
#define rep(i,a,n) for(int i=a;i<=n;i++) 
typedef long long LL;
const int maxn=1e2+10;
const int N=1e4+10;
const int mod=20201114;
int a[maxn];
int main(){
    std::ios::sync_with_stdio(false);
    a[1]=1;
    for(int i=1;i<=N;i++){
    	int x=0;
    	for(int j=1;j*j<=i;j++){
    		if(j*j==i) x+=1;
    	    else if(i%j==0) x+=2;
		}
		for(int j=1;j<=x;j++) if(!a[j]) a[j]=i;
	}
    int ans=0;
	for(int i=1;i<=60;i++) ans+=a[i];
	cout<<ans<<endl;
	return 0;
}
