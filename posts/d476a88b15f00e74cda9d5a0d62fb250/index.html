本题要求编写程序，计算序列 1!+2!+3!+⋯ 的前N项之和。输入格式:输入在一行中给出一个不超过12的正整数N。输出格式:在一行中输出整数结果。输入样例:5输出样例:153

#include<stdio.h>
double factorial(int n)
{
    double sum=1;
    for(int a=1;a<=n;a++)
    {
          sum=sum*a;
    }
    return sum;
}
int main()
{
    int N,result=0,i;
    scanf("%d",&N);
    if(N<=12)
    {
        for(i=1;i<=N;i++){
            result=result+factorial(i);
        }
    }
    printf("%d",result);
    return 0;
}
