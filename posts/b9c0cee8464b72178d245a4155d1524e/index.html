curl 模拟表单发送

    // 初始化一个curl
    $ch = curl_init();
    // 设置相应的会话选项
    $url = "http://localhost/index2.php";
    $postDate = array(
        'title' => '哈哈哈',
        'content' => '今天是周五',
        );
    // 设置提交的网址
    curl_setopt($ch,CURLOPT_URL,$url);
    // 设置数据提交方式
    curl_setopt($ch,CURLOPT_POST,1);
    // 设置数据
    curl_setopt($ch,CURLOPT_POSTFIELDS,$postDate);
    curl_exec($ch);

    curl_close($ch);
