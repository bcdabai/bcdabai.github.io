// 定时器开启(通过某种方式触发定时器)
    openTimer() {
      this.isProgress = true;
      // 进度慢
      timer1 = setInterval(() => {
        this.percentage++;
        if (this.percentage > 85 && this.percentage <= 100) {
          if (timer1) {
            clearInterval(timer1);
          }
          // 进度快
          timer2 = setInterval(() => {
            this.percentage++;
            if (this.percentage > 100) {
              clearInterval(timer2);
              // 加载完成,进度条消失
              this.isProgress = false;
              this.percentage = 0;
            }
          }, 50);
        }
      }, 200);
    }
