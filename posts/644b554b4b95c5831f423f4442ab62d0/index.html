C#学习 winform实现简单音乐播放器

窗体设计

代码实现

using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.IO;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;
using WMPLib;

namespace _31_MusicPlayer
{
    public partial class Form1 : Form
    {
        //声明一个list，用来存储文件的路径
        List<string> urlList = new List<string>();

        double max, min;

        public Form1()
        {
            InitializeComponent();
        }

        //添加歌曲
        private void btnInput_Click(object sender, EventArgs e)
        {
            //实例化一个打开文件的对话框
            OpenFileDialog ofd = new OpenFileDialog();
            //让选择器可以同时选择多个文件
            ofd.Multiselect = true;
            ofd.Title = "请选择音乐文件";
            //指定选择文件的类型
            ofd.Filter = "(*.mp3)|*.mp3";
            if(ofd.ShowDialog() == DialogResult.OK)
            {
                //把用户选择的文件存储到数组中
                string[] nameList = ofd.FileNames;
                //读取数组中的数据
                foreach(string url in nameList)
                {
                    //Path.GetFileNameWithoutExtension()获取不具有扩展名的文件名
                    listBoxMusics.Items.Add(Path.GetFileNameWithoutExtension(url));
                    urlList.Add(url);
                }


            }

        }

        //列表选择变化时，播放选中的歌曲
        private void listBoxMusics_SelectedIndexChanged(object sender, EventArgs e)
        {
            //ListBox中的索引和urlList中的索引相对应
            //获取当前选中歌曲的索引
            int selectIndex = listBoxMusics.SelectedIndex;
            if(listBoxMusics.SelectedItem != null)
            {
                lblMusicName.Text = listBoxMusics.SelectedItem.ToString();

                //把存储的url地址赋给播放器控件
                axWindowsMediaPlayer1.URL = urlList[selectIndex];
                timer1.Enabled = true;
            }
           
        }

        //下一首
        private void btnNext_Click(object sender, EventArgs e)
        {
            //ListBox中的索引和urlList中的索引相对应
            //获取当前选中歌曲的索引-1
            int selectIndex = listBoxMusics.SelectedIndex + 1;

            selectIndex = selectIndex == listBoxMusics.Items.Count ? 0 : selectIndex;//listBoxMusics.SelectedIndex

            //更新选中行
            listBoxMusics.SelectedIndex = selectIndex;
            //把存储的url地址赋给播放器控件
            axWindowsMediaPlayer1.URL = urlList[selectIndex];
            lblMusicName.Text = listBoxMusics.SelectedItem.ToString();
        }

        //上一首
        private void btnFront_Click(object sender, EventArgs e)
        {
            int selectIndex = listBoxMusics.SelectedIndex - 1;
            //listBoxMusics.Items.Count 长度
            selectIndex = selectIndex < 0 ? listBoxMusics.Items.Count-1 : selectIndex;

            //更新选中行
            listBoxMusics.SelectedIndex = selectIndex;
            //把存储的url地址赋给播放器控件
            axWindowsMediaPlayer1.URL = urlList[selectIndex];
            lblMusicName.Text = listBoxMusics.SelectedItem.ToString();
        }
        //播放
        private void btnPlay_Click(object sender, EventArgs e)
        {
            if(listBoxMusics.Items.Count == 0)
            {
                MessageBox.Show("未发现任何音乐");
            }
            else
            {
                int selectIndex = listBoxMusics.SelectedIndex;
                if (selectIndex < 0)//无选中音乐
                {
                    //判断列表中是否有选中的歌曲
                    selectIndex = selectIndex < 0 ? 0 : selectIndex;

                    //更新选中行
                    listBoxMusics.SelectedIndex = selectIndex;
                    //把存储的url地址赋给播放器控件
                    axWindowsMediaPlayer1.URL = urlList[selectIndex];
                    lblMusicName.Text = listBoxMusics.SelectedItem.ToString();
                }
                else
                {
                    axWindowsMediaPlayer1.Ctlcontrols.play();
                }
                timer1.Enabled = true;
            }
           
        }

        //暂停
        private void btnPause_Click(object sender, EventArgs e)
        {
            axWindowsMediaPlayer1.Ctlcontrols.pause();
        }
        //停止
        private void btnStop_Click(object sender, EventArgs e)
        {
            axWindowsMediaPlayer1.Ctlcontrols.stop();
            timer1.Enabled=false;
        }

        //鼠标按下
        private void trackBar1_MouseDown(object sender, MouseEventArgs e)
        {
            //鼠标按下时，获取播放时的位置
            timer1.Enabled = false;
            axWindowsMediaPlayer1.Ctlcontrols.pause();
        }
        //鼠标抬起
        private void trackBar1_MouseUp(object sender, MouseEventArgs e)
        {
            Double doValue = trackBar1.Value;//获取被拖动以后的位置
            axWindowsMediaPlayer1.Ctlcontrols.currentPosition = doValue;//重置播放位置
            axWindowsMediaPlayer1.Ctlcontrols.play();
            timer1.Enabled = true;
        }

        //timer
        private void timer1_Tick(object sender, EventArgs e)
        {
            //获取音乐的时间长度
            max = axWindowsMediaPlayer1.currentMedia.duration;
            //获取当前歌曲的播放位置
            min = axWindowsMediaPlayer1.Ctlcontrols.currentPosition;

            trackBar1.Maximum = (int)(max);
            trackBar1.Value = (int)(min);

            if(axWindowsMediaPlayer1.playState == WMPPlayState.wmppsStopped)
            {
                //一首歌播放结束获取下一首歌曲索引
                int selectIndex = listBoxMusics.SelectedIndex + 1;
                //最后一首
                selectIndex = selectIndex == listBoxMusics.Items.Count ? 0 : selectIndex;

                axWindowsMediaPlayer1.URL = urlList[selectIndex];
                listBoxMusics.SelectedIndex = selectIndex;
                lblMusicName.Text = listBoxMusics.SelectedItem.ToString();
                trackBar1.Value = 0;
                timer1.Enabled = true;
            }
        }

        //xml数据库
    }
}

点击添加歌曲选择本地下载的mp3歌曲文件.
