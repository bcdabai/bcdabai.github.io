结构体数组：有5个同学的信息（包括：学号，名字，成绩），要求按照成绩从大到小的顺序输出

#include<stdio.h>
int main()
{
	struct student{
		int no;
		char name[20];
		int cj;
	}a[5]={{01,"zhao",78},{02,"hu",76},{03,"wang",98},{04,"li",34},{05,"ma",44}};
	int i,j;
	struct student t;
	for(i=0;i<5;i++)
	{
		j=0;
		for(j=0;j<4;j++)
		{
			if(a[j].cj<a[j+1].cj)
			{t=a[j];a[j]=a[j+1];a[j+1]=t;
			}
		}
	}
	for(i=0;i<5;i++)
	{
		printf("%d %s %d\n",a[i].no,a[i].name,a[i].cj);
		
	}
}
