查看docker中的mysql版本，查看docker中容器,查看mysql版本
# 查看docker中容器 （NAMES）
docker ps
#    CONTAINER ID  ....  NAMES
#    a00000000000  ....  mysql8

# 进入mysql容器
docker exec -it mysql8 bash

# 登录mysql,输入账号回车填写密码登录
mysql -uroot -p
#    Server version: 8.0.21 MySQL Community Server - GPL
