        },
        // more data items
      ],
    };
  },
  methods: {
    toggleShowMore(index) {
      this.data[index].showMore = !this.data[index].showMore;
    },
    getAuthors(authors, showMore) {
      if (authors.length > 5 && !this.showMore) {
        return authors.slice(0, 5);
      }
      return authors;
    },
  },
};
