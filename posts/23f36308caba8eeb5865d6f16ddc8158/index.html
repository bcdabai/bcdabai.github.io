去除注释代码如下：

/**

* Created by PhpStorm.

function removeComment($content)

{

return preg_replace("/(\/\*(\s|.)*?\*\/)|(\/\/.(\s|.*)\n)|(#(\s*)?(.*))/", ‘‘, str_replace(array("\r\n", "\r"), "\n", $content));

}

$content = file_get_contents("./test.php");

echo removeComment($content);
