Element ui Table 嵌套单选多选 阻止@row-click事件冒泡


直接贴代码，主要是添加 @click.native.stop 

<el-table-column
    label="吃饭吗"
    height="100%"
    width="140"
    align="center">
    <template slot-scope="scope">
      <el-radio-group v-model="scope.row.calculateScore" @click.native.stop>
        <el-radio label="10">吃</el-radio>
        <el-radio label="20">不吃</el-radio>
      </el-radio-group>
    </template>
</el-table-column>
