ajax

干货警告！在ajax回调函数里设置更新的那部分配置即可

$.ajax({
        url: "http://www.example.com",
        type: "POST",
        success: function (data) {
            myChart.setOption({
                series: [{
                    axisLine: {
                        lineStyle: {
                            width: 3,
                            color: [
                                [0.8, '#ff802c'],
                            ]
                        }
                    },
                    data: [{
                        value: 80
                    }]
                }]
            });
        }
    });

    option = {
        backgroundColor: '',

            top: 10,
        },
