用JavaScript判断一个整数，属于哪个范围：大于0；小于0；等于0

<script>

        // 判断一个整数，属于哪个范围：大于0；小于0；等于0

        var num = prompt('请您输入一个整数');

        if (num > 0) {

            alert(`您输入的${num}大于0`);

        } else if (num < 0) {

            alert(`您输入的${num}小于0`);

        } else {

            alert(`您输入的${num}等于0`);

        }

    </script>
