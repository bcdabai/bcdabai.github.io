给定一个带头节点的单链表，所有数据值无序，删除介于min和max的所有值

#include <stdio.h>
#include <math.h>

typedef struct LNode{
    int data;
    struct LNode *next;
}LNode,*LinkList;
void RangeDelete(LinkList &L,int min,int max){
    LNode *p=L->next;
    LNode *pre=L;
    while(p!=NULL){
        if(p->data>=min&&p->data<=max){
            pre=p->next;
            printf("%d",p->data);
            free(p);
            p=pre;
        }
        else{
            pre=p;
            p=p->next;
        }
    }
}

void input(LNode *head)
{
    printf("\nINPUT\n");
    LNode *p;
    p=head;/*指向表头*/
    int a;
    for(int i=0;i<=5;i++)
    {
        scanf("%d",&a);
        if(a==-1)
            return ;
        LNode *q=new LNode ;
        q->data=a;
        q->next=NULL;
        p->next=q;
        p=q;
    }
}
int main()
{
    LNode *L=new LNode;
    L->next=NULL;
    input(L);
    int min1,max1;
    scanf("%d%d",&min1,&max1);
    RangeDelete(L,min1,max1);
    return 0;
}
