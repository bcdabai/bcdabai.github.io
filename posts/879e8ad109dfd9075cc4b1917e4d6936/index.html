作业--数组

package sz;

public class sz1 {
    public static void main(String[] args) {
        int[] a= {-1,-2,-4,-8,-4,-7,-1,-5};
        System.out.println(max(a));
    }
    public static int max(int[] array) {
        int n=array.length;
        int max=0;
        int maxsum=array[0];
        for(int i = 0;i < n;i++)
        {
            if (max <= 0) {
                max = array[i];
            }else {
                max += array[i];
            }
            
            if (maxsum < max) {
                maxsum = max;
            }
        }
        return maxsum;
}
}
