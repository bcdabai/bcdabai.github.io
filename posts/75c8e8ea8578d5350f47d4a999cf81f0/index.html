import csv

iplist=[]

def ReadCsvFileFirst(iplist):
    with open('./test.csv','r',encoding='utf-8-sig') as f:
        reader = csv.reader(f)
        for row in reader:
            # 打印主机Ip
            for hostip in row[0:1]:
                if hostip == "主机名":
                    continue
                else:
                    ListAdd(iplist,hostip)
                    # print(hostip,end="\t")
        return iplist
