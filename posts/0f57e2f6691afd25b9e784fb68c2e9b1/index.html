用Java程序输出1+2+3+···+100的和（三种循环语句）

//while循环
public class Test {
	public static void main(String[] args){
		int i=1;
		int sum=0;
		while(i<=100){
			sum=sum+i;
			i++;
		}
		System.out.println(sum);
	}
	
}


//for循环
public class Test {
	public static void main(String[] args){
		int sum=0;
		for(int i=1;i<=100;i++){
			sum=sum+i;
		}
		System.out.println(sum);
	}
	
}



//do while循环
public class Test {
	public static void main(String[] args){
		int i=0;
		int sum=0;
		do{
			i++;
			sum=sum+i;
		}while(i<100);
		System.out.println(sum);
	}
	
}
