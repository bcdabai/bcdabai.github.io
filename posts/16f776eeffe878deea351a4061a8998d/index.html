雷达距离分辨率和速度分辨率计算

antenna = 1; % 天线数
C = 300000000; % m/s
fadc = 10; % M

nFrame = 200; % 帧数
chirp = 128; % 一帧的chirp数
point = 256; % 一chirp中的点数
B = 400; % Mhz
T = 100; % us
PRT = 300; % us

k = B/T;% Mhz/us 斜率
centerB = 24 + B/(2*1000);% GHz 中心频率

Range_r = C/(2*k*point/(fadc*1000) *1000000000); % 距离分辨率
Doppler_r = C/(2*chirp*antenna*centerB*PRT*1000); % 速度分辨率
