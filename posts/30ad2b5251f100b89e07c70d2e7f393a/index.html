SQLAlchemy中一对一，一对多，多对一Mode转dict

def sqlModeToDict(sqlObj):
    resuDict = {}
    for k,v in sqlObj.__dict__.items():
        if isinstance(v, list):
            info = sqlModeToDict(v[0])
            resuDict.update(info)
        else:
            resuDict[k] = v
    resuDict.pop('_sa_instance_state',None)
    return resuDict
