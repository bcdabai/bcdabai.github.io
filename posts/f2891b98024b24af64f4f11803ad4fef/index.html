【案例5-3】 模拟用户登录

Test类

package anli5_3;

import java.util.Scanner;

//模拟用户登录
public class Test {
    public static void main(String[] args) {
        String username="name";
        String password="mima";

        Scanner sc=new Scanner(System.in);

        for (int i = 1; i <= 3; i++) {
            System.out.println("请输入用户名");
            String scU= sc.nextLine();
            System.out.println("请输入密码");
            String scP=sc.nextLine();

            if (username.equals(scU)&&password.equals(scP)){
                System.out.println("登陆成功");
                break;
            }else {
                if (3-i==0){
                    System.out.println("你的登录次数以用完，请与管理员联系");
                }else {
                    System.out.println("登陆失败，您还剩余"+(3-i)+"次机会");
                }
            }

        }
    }
}
