1、首页index.jsp

<%@ page language="java" contentType="text/html; charset=utf-8"
    pageEncoding="utf-8"%>
<!DOCTYPE html>
<html>
<head>
<meta charset="utf-8">
<title>index</title>
</head>
<body>
	<%
	//接收是否登录成功数据
	String flag=request.getParameter("flag");
	if(flag==null){
		flag="";
	}
	if(flag.equals("sucessful")){
		String okname=(String)request.getAttribute("okname");
		%>
		<%=okname %>你好!/<a href="index.jsp">注销</a>
		<% 
	}else{
		%>
		<a href="login.jsp">登录</a>/
		<a href="reg.jsp">注册</a>
		<% 
	}
	%>
	
</body>
</html>

3、注册成功，显示注册信息，showReg.jsp

<%@ page language="java" contentType="text/html; charset=utf-8"
    pageEncoding="utf-8"%>
<!DOCTYPE html>
<html>
<head>
<meta charset="utf-8">
<title>展示注册信息</title>
</head>
<body>
	<%
	request.setCharacterEncoding("utf-8");
	//接受数据
	String name=request.getParameter("username");
	String pwd=request.getParameter("password");
	String okPwd=request.getParameter("okPassword");
	String sex=request.getParameter("sex");
	String[] interest=request.getParameterValues("interest");
	//验证数据
	if(name==null||name.equals("")||pwd==null||pwd.equals("")||okPwd==null||okPwd.equals("")||sex==null||sex.equals("")||interest==null||interest.equals("")){
		%>
		<jsp:forward page="reg.jsp">
		<jsp:param value="页面上的内容不能为空！" name="mess"/>
		</jsp:forward>
		<%
	}
	else if(!pwd.equals(okPwd)){
		%>
		<jsp:forward page="reg.jsp">
		<jsp:param value="两次输入的密码不一致！请重新输入" name="mess"/>
		</jsp:forward>
		<%
	}
	
	//处理数据
	//将String[]->String
	String strHobby="";
	for(int i=0;i<interest.length;i++){
		strHobby+=interest[i]+",";
	}
	
	//去掉末尾的,
	if(!strHobby.equals("")){
		strHobby=strHobby.substring(0, strHobby.length()-1);
	}
	
	//在数据库添加数据
	
	//在页面显示数据
	%>
	恭喜您注册成功！以下为您的注册信息：
	<br>
	用户名:<%=name %>
	<br>
	密码:<%=pwd %>
	<br>
	性别:<%=sex %>
	<br>
	兴趣:<%=strHobby%>
	
	
</body>
</html>

5、登录判断，loginProcess.jsp

<%@ page language="java" contentType="text/html; charset=utf-8"
    pageEncoding="utf-8"%>
<!DOCTYPE html>
<html>
<head>
<meta charset="utf-8">
<title>loginProcess</title>
</head>
<body>
	<form action="index.jsp">
	<%!
	String flag="fail";
	%>
	<%
	request.setCharacterEncoding("utf-8");
	//接收数据
	String name=request.getParameter("username");
	String pwd=request.getParameter("password");
	//验证数据
	if(name==null||name.equals("")||pwd==null||pwd.equals("")){
		%>
		<jsp:forward page="login.jsp">
		<jsp:param value="用户名或密码不能为空！" name="mess"/>
		</jsp:forward>
		<%
	}else if(name.length()>15){
		%>
		<jsp:forward page="login.jsp">
		<jsp:param value="用户名长度不能超过15" name="mess"/>
		</jsp:forward>
		<%
	}else{
		flag="sucessful";
		request.setAttribute("okname", name);
		%>
		<jsp:forward page="index.jsp">
			<jsp:param name="flag" value="sucessful"/>
		</jsp:forward>
		<%
		
	}
	//处理数据
	%>
	</form>
</body>
</html>
