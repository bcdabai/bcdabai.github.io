#include <stdio.h>

int main()
{
    double sum=0;
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        long jec=1;
        for(int j=1;j<=i;j++)
        {
            jec*=j;
        }
        sum+=1.0/jec;
    }
    printf("%.2lf",sum);
    return 0;
}
