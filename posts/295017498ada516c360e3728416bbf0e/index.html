实现 当 a ＞ b 时交换两数的值（目的是为了让第一个变量永远是两个变量中较小的），要求不使用第三个变量

public class Main {
    public static void main(String[] args) {
        int a = 8, b = 1;
        a = a > b ? a ^ b ^ (b = a) : a;
        // a = a > b ? a + b - (b = a) : a; // 也可
        System.out.println(a + " " + b);
    }
}
