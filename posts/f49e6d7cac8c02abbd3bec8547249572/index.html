3、代码实现（c语言）

#include <stdio.h>

int main()
{
   int i,j,n,space;
   printf("请输入树的层数：");
   scanf("%d",&n);
   for(i=1;i<=n;i++)
   {
       for(j=1;j<=i+1;j++)
       {
           for(space=1;space<=n+1-j;space++)
           {
               printf(" ");
           }
           for(space=1;space<=2*j-1;space++)
           {
               printf("*");
           }
           printf("\n");
       }
   }
   for(space=1;space<=n+1;space++)
   {
       if(space==n+1)
       {
       printf("|");
       continue;
       }
       printf(" ");
   }
   printf("\n");
   for(space=1;space<=2*n+1;space++)
   {
       if(space==n+1)
       {
       printf("V");
       continue;
       }
       printf("=");
   }
   return 0;
}

4、代码实现（python）

n = int(input("请输入树的层数："))
for i in range(1,n+1):
    for j in range(1,i+2):
        for k in range(1,n+2-j):
            print(" ", end = "")
        for k in range(1,2*j):
            print('*',end = "")
        print("")

for space in range(1,n+2):
    if space == n+1:
        print("|",end = "")
    else:
        print(" ",end = "")

print("")
for space in range(1,2*n+2):
    if space==n+1:
        print("V",end = "")
    else:
        print("=",end = "")
