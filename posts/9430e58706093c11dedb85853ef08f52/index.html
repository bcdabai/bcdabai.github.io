用指针方法编写一个程序，输入3个整数，将它们按由小到大的顺序输出

#include <stdio.h>

void swap(int *pa,int *pb)

{

int temp;

temp = *pa;

*pa = *pb;

*pb = temp;

}

void main()

{

int a,b,c,temp;

scanf("%d%d%d",&a,&b,&c);

if(a>b)

swap(&a,&b);

if(b>c)

swap(&b,&c);

if(a>c)

swap(&a,&c);

printf("%d,%d,%d",a,b,c);

}
