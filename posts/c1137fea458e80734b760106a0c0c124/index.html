MybatisPlus按时间段进行查询

String beginTime = "2024-01-23";
String endTime = "2024-01-23";
 
QueryWrapper<IotDevice> queryWrapper = new QueryWrapper<>();
queryWrapper.select().apply(StrUtil.isNotBlank(beginTime),
                        "date_format(create_date,'%Y-%m-%d') >= date_format('" + beginTime+ "','%Y-%m-%d')")
                .apply(StrUtil.isNotBlank(endTime),
                        "date_format(create_date,'%Y-%m-%d') <= date_format('" + endTime+ "','%Y-%m-%d')");
List<IotDevice> iotDeviceList= iotDeviceService.list(queryWrapper);
